\begin{frame}[fragile]\frametitle{The Three Waves of Modern AI}
\begin{itemize}
\item \textbf{Wave 1: Predictive AI (2010-2021)}
    \begin{itemize}
    \item Classification, Regression, Clustering
    \item Traditional ML: Random Forests, XGBoost, Neural Networks
    \item Use case: Predict customer churn, fraud detection
    \end{itemize}
\item \textbf{Wave 2: Generative AI (2022-2024)}
    \begin{itemize}
    \item Text generation (GPT), Image generation (DALL-E, Midjourney)
    \item Foundation models trained on internet-scale data
    \item Use case: Content creation, code generation, chatbots
    \end{itemize}
\item \textbf{Wave 3: Agentic AI (2024+)}
    \begin{itemize}
    \item Autonomous decision-making, multi-step reasoning
    \item AI agents that can plan, execute, and learn
    \item Use case: AI employees, research assistants, autonomous systems
    \end{itemize}
\end{itemize}

{\tiny (Ref: Sequoia Capital AI Market Analysis 2024)}
\end{frame}

\begin{frame}[fragile]\frametitle{What is Data Science?}

\begin{itemize}
\item Science of Data (obviously)
\item Use of Data for Applications
\item Some parts of AI uses Data to find patterns and insights which are helpful in multiple applications
\item Machine and Deep Learning that part of AI that leverages data.
\end{itemize}

So, more on AI-ML soon \ldots

\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{What is a Data Scientist?}
\begin{itemize}
\item Software + Mathematics + ML/DL specific techniques
\item Discovers patterns and trends in datasets to get insights.
\item Creates forecasting algorithms like Classification, Clustering, etc
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Sample comparison: Business Intelligence vs Data Science}
\begin{table}[]
\begin{tabular}{p{0.4\textwidth}p{0.4\textwidth}}
\hline
{\bf Business Intelligence} & {\bf Data Science}         	\\ \hline
Extract insights using past data    & Predict future using past data       		\\ \hline
Uses structured data      & Uses both structured and unstructured data  				\\ \hline
Use of basic statistics with emphasis on visualization (dashboards, reports)    & Leverages more sophisticated statistical and predictive analysis and machine learning (ML)            	\\ \hline
\end{tabular}
