\begin{frame}[fragile]\frametitle{The Imperative: Become AI-Enabled}
\begin{itemize}
  \item AI is not optional, it is the new baseline literacy
  \item \textbf{Engineering or non-engineering, science or non-science}, domain does not matter
  \item The first step for everyone: become \textbf{AI-enabled}
    \begin{itemize}
      \item Know what AI tools exist and what they can do
      \item Use AI to amplify your existing expertise, not replace it
      \item Treat AI as a collaborator, not a threat
    \end{itemize}
  \item The next level: become \textbf{AI-native}
    \begin{itemize}
      \item Default to AI-first thinking in every workflow
      \item Build, iterate, and solve problems with AI at the center
    \end{itemize}
  \item ``You will not lose your job to AI, you will lose it to a person who uses AI''
  \item Start today: pick one tool, one use case, one week of daily practice
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{The Three Waves of Modern AI}

\begin{columns}
    \begin{column}[T]{0.6\linewidth}
\begin{itemize}
\item \textbf{Wave 1: Predictive AI (2010-2021)}
    \begin{itemize}
    \item Classification, Regression, Clustering
    \item Traditional ML: Random Forests, XGBoost, Neural Networks
    \item Use case: Predict customer churn, fraud detection
    \end{itemize}
\item \textbf{Wave 2: Generative AI (2022-2024)}
    \begin{itemize}
    \item Text generation (GPT), Image generation (DALL-E, Midjourney)
    \item Foundation models trained on internet-scale data
    \item Use case: Content creation, code generation, chatbots
    \end{itemize}
\item \textbf{Wave 3: Agentic AI (2024+)}
    \begin{itemize}
    \item Autonomous decision-making, multi-step reasoning
    \item AI agents that can plan, execute, and learn
    \item Use case: AI employees, research assistants, autonomous systems
    \end{itemize}
\end{itemize}

\begin{frame}[fragile]\frametitle{Forward Deployed Engineer (FDE): The Crucial Role}
\begin{itemize}
  \item An FDE is \textbf{AI-native by default}, fluent in tools, models, and APIs
  \item Deep domain expertise in their original discipline (engineering, law, medicine, finance \ldots)
  \item \textbf{Deployed at the client side}, embedded within the client's team, not behind a desk
  \item Solves \textit{real, domain-specific} problems using AI:
    \begin{itemize}
      \item \textbf{Banking:} Automate risk assessment, fraud pipelines, KYC workflows
      \item \textbf{Legal:} Contract analysis, case research, compliance checks via LLMs
      \item \textbf{Pharmaceutical:} Drug interaction prediction, clinical trial data mining
      \item \textbf{Manufacturing:} Predictive maintenance, defect detection with vision AI
    \end{itemize}
  \item Bridge between the \textit{AI product} and the \textit{domain expert}, speaks both languages
  \item High trust, high ownership: often the \textbf{only technical AI voice} on a client engagement
  \item Pioneered by companies like Palantir; now spreading across every industry vertical
\end{itemize}
{\tiny (Ref: Palantir FDE Model; a16z ``The New AI Stack'' 2024)}
\end{frame}

\begin{frame}[fragile]\frametitle{Your Path: Aim to Be the FDE}
\begin{itemize}
  \item Whatever your background, \textbf{that is your domain moat}
  \item Layer AI-enablement on top: tools, prompting, basic model intuition
  \item Progress toward AI-native: build small AI-powered workflows in your field
  \item \textbf{Target the FDE archetype} in your domain:
    \begin{itemize}
      \item Civil engineer + AI $\rightarrow$ FDE for smart infrastructure firms
      \item HR professional + AI $\rightarrow$ FDE for talent-tech companies
      \item Doctor + AI $\rightarrow$ FDE for health-tech / diagnostics startups
    \end{itemize}
  \item FDEs command \textbf{premium compensation} and deep client relationships
  \item The role is scarce today, which means the window to enter is \textbf{now}
  \item \textbf{Next:} We explore AI fundamentals, tools, techniques, and concepts to begin this journey
\end{itemize}
\end{frame}
