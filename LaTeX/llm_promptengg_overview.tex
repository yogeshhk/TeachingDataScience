\begin{frame}[fragile]\frametitle{Text Summarization}
\begin{itemize}
    \item \textbf{Task:} Condense a long piece of text into a short, clear summary.
    \item \textbf{How:} Paste the text and tell the AI how short the summary should be.
\end{itemize}
\begin{lstlisting}[language=HTML]
Prompt:
Antibiotics are a type of medication used to treat bacterial 
infections. They work by either killing the bacteria or 
preventing them from reproducing, allowing the body's immune 
system to fight off the infection. They are not effective 
against viral infections, and using them inappropriately 
can lead to antibiotic resistance.
Explain the above in one sentence:

Output:
Antibiotics treat bacterial infections by killing or 
stopping bacteria from reproducing, but don't work on 
viruses, and overuse causes resistance.
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Information Extraction}
\begin{itemize}
    \item \textbf{Task:} Pull out a specific piece of information from a larger block of text — like a search inside a document.
    \item \textbf{How:} Give the AI the text and tell it exactly what to look for.
\end{itemize}
\begin{lstlisting}[language=HTML]
Prompt:
Author-contribution statements in research papers should 
state clearly whether authors used AI technologies such as 
ChatGPT in the preparation of their manuscript...

Mention the large language model based product mentioned 
in the paragraph above:

Output:
The large language model based product mentioned is ChatGPT.
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Question Answering}
\begin{itemize}
    \item \textbf{Task:} Answer a specific question using only the context provided — the AI does not guess; it reads what you gave it.
    \item \textbf{How:} Give context + question. Ask it to say ``Unsure'' if the answer is not in the context.
\end{itemize}
\begin{lstlisting}[language=HTML]
Answer the question based on the context below. 
Keep the answer short. Respond "Unsure about answer" 
if not sure about the answer.

Context: Teplizumab traces its roots to a New Jersey drug 
company called Ortho Pharmaceutical. There, scientists 
generated an early version of the antibody, dubbed OKT3.

Question: What was OKT3 originally sourced from?

Output:
Mice.
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Code Generation}
\begin{itemize}
    \item \textbf{Task:} Generate working code just from a plain-English description — no programming knowledge needed to describe \emph{what} you want.
\end{itemize}
\begin{lstlisting}[language=HTML]
Prompt:
/*
Ask the user for their name and say "Hello"
*/
Output:
let name = prompt("What is your name?");
console.log(`Hello, ${name}!`);

--------------------------------------------------
Prompt:
Table departments, columns = [DepartmentId, DepartmentName]
Table students, columns = [DepartmentId, StudentId, StudentName]
Create a MySQL query for all students in Computer Science

Output:
SELECT StudentId, StudentName FROM students 
WHERE DepartmentId IN 
  (SELECT DepartmentId FROM departments 
   WHERE DepartmentName = 'Computer Science');
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Reasoning — and Where It Can Go Wrong}
\begin{itemize}
    \item LLMs can do arithmetic but they sometimes get it wrong because they are \textbf{predicting likely text}, not computing.
    \item \textbf{Fix:} Break the problem into steps in your prompt.
\end{itemize}
\begin{lstlisting}[language=HTML]
Prompt (naive):
What is 9,000 * 9,000?
Output: 81,000,000   [correct here, but unreliable on harder sums]

--------------------------------------------------
Prompt (step-by-step):
The odd numbers in this group add up to an even number: 
15, 32, 5, 13, 82, 7, 1. 
Solve by breaking the problem into steps. First, identify 
the odd numbers, add them, and indicate whether the result 
is odd or even.

Output:
Odd numbers: 15, 5, 13, 7, 1
Sum: 41 
41 is an odd number.
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Calling a Prompt Programmatically (Python)}

\begin{lstlisting}[language=Python]
# Requires: pip install openai python-dotenv
from openai import OpenAI
import os
from dotenv import load_dotenv, find_dotenv

load_dotenv(find_dotenv())
client = OpenAI(api_key=os.getenv('OPENAI_API_KEY'))

def get_completion(prompt, model="gpt-4o"):
    response = client.chat.completions.create(
        model=model,
        messages=[{"role": "user", "content": prompt}],
        temperature=0,  # 0 = deterministic, 1 = creative
    )
    return response.choices[0].message.content

# Example usage
result = get_completion("Explain photosynthesis in one sentence.")
print(result)
\end{lstlisting}

{\tiny Note: Uses the modern \texttt{openai} SDK (v1.0+). The legacy \texttt{ChatCompletion.create()} API is deprecated.}
		
\end{frame}
