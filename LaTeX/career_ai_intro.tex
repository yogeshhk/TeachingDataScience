\begin{frame}[fragile]\frametitle{Paradigm Shift: Foundation Models}
\begin{itemize}
\item \textbf{Old Way (Pre-2022):} Train a specific model for a specific task (e.g., sentiment analysis model, translation model). Requires massive labeled data.
\item \textbf{New Way (Post-2022):} Train one massive "Foundation Model" (e.g., GPT-4, Llama 3) on the whole internet.
\item \textbf{Adaptation:} Fine-tune or "Prompt" this foundation model for almost ANY task.
\end{itemize}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Paradigm Shift: Foundation Models}
\begin{columns}
    \begin{column}[T]{0.52\linewidth}
\textbf{Old Way (Pre-2022):}
\begin{itemize}
\item Train specific model for specific task
\item Sentiment analysis $\rightarrow$ Train model
\item Translation $\rightarrow$ Train model
\item Summarization $\rightarrow$ Train model
\item Requires massive labeled datasets
\item Months of training time
\end{itemize}

\textbf{The Problem:}
\begin{itemize}
\item Expensive (\$100K-\$1M+ per model)
\item Time-consuming (3-12 months)
\item Requires ML expertise
\item Hard to iterate
\end{itemize}
    \end{column}
    \begin{column}[T]{0.48\linewidth}
\textbf{New Way (2022+):}
\begin{itemize}
\item One massive "Foundation Model"
\item GPT-4, Claude, Gemini, Llama
\item Trained on entire internet
\item Can do \textbf{ANY} task via prompting
\item Fine-tune only when needed
\end{itemize}

\textbf{The Impact:}
\begin{itemize}
\item Cost: \$0.01-\$10 per use case
\item Time: Hours to days (not months)
\item No ML expertise needed initially
\item Rapid iteration via prompts
\end{itemize}

    \end{column}
\end{columns}

\vspace{0.3cm}
\textbf{Key Insight:} 80\% of "AI projects" now mean "API calls to foundation models"

{\tiny (Ref: Andreessen Horowitz State of AI 2024)}
\end{frame}
