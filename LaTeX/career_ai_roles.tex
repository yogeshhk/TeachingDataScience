\begin{frame}[fragile]\frametitle{LLM Ops Engineer (ML Ops 2.0)}
\begin{columns}
\begin{column}[T]{0.5\linewidth}
\textbf{What Changed from ML Ops:}
\begin{itemize}
\item Less about model training pipelines
\item More about LLM deployment at scale
\item Focus on inference optimization
\item Cost management (inference is expensive!)
\item Quality monitoring (hallucinations)
\item Implement caching strategies (reduce API costs 50-90\%)
\end{itemize}
\end{column}
\begin{column}[T]{0.5\linewidth}
\textbf{Core Responsibilities:}
\begin{itemize}
\item Deploy and serve LLMs (OpenAI, Anthropic, self-hosted)
\item A/B test different models for cost/quality tradeoffs
\item Monitor latency, costs, and output quality
\item Build evaluation pipelines (automated testing)
\item Manage vector databases and embedding workflows
% \item Handle model fallbacks and rate limiting
\end{itemize}

    \end{column}
\end{columns}

\begin{frame}[fragile]\frametitle{AI Product Manager: Bridge between business and AI }


\begin{columns}
    \begin{column}[T]{0.6\linewidth}
\textbf{Core Skills:}
\begin{itemize}
\item Understanding of LLM pros \& cons
\item Prompt engineering (hands-on experience)
\item Product strategy for AI applications
\item Understanding of AI costs and scaling
\item User research for AI interactions
\item Ethical AI and bias mitigation
\end{itemize}

\textbf{Typical Responsibilities:}
\begin{itemize}
\item Define product requirements for AI
\item Design conversational UI/UX flows
\item Evaluate model choices 
\item Balance cost vs quality tradeoffs
\item Measure AI feature success metrics
\item Coordinate between stakeholders
\end{itemize}
    \end{column}
    \begin{column}[T]{0.4\linewidth}


\textbf{Success Examples:}
\begin{itemize}
\item Perplexity: AI search
\item Notion AI: Writing assistant
\item Cursor: AI code editor
\item Harvey: Legal AI
\end{itemize}

\textbf{Salary Range:}
\begin{itemize}
\item India: Rs 30-60 LPA
\item US: \$180K-\$280K
\end{itemize}

\textbf{Entry Path:}
Best for domain experts who want to transition without deep coding. Need to build AI prototypes but not production systems.

{\tiny (Ref: Lenny's Newsletter - AI PM Deep Dive )}
    \end{column}
\end{columns}

\begin{frame}[fragile]\frametitle{The AI Engineer: Build AI-powered applications }

\textbf{Core Skills:}
\begin{itemize}
\item Prompt Engineering \& Chain-of-Thought reasoning
\item RAG (Retrieval Augmented Generation) implementation
\item Vector databases (Pinecone, Weaviate, Chroma)
\item LLM orchestration (LangChain, LlamaIndex)
\item API integration (OpenAI, Anthropic, Google)
\item Evaluation frameworks (LLM-as-a-Judge, RAGAS)
\end{itemize}

\textbf{Typical Day:}
\begin{itemize}
\item Design prompt chains for complex workflows
\item Build RAG systems for knowledge bases
\item Optimize context windows and token usage
\item A/B test different models for cost/quality
\item Debug hallucinations and edge cases
\end{itemize}

\begin{frame}[fragile]\frametitle{Prompt Engineer / AI Whisperer}
Design, test, and optimize prompts for complex AI workflows.

\begin{columns}
\begin{column}[T]{0.6\linewidth}

\textbf{Core Skills:}
\begin{itemize}
\item Advanced prompting techniques (CoT, ToT, ReAct)
\item Systematic prompt testing and evaluation
\item Building reusable prompt libraries
\item Understanding model limitations and behaviors
\item Creating few-shot examples
\item Adversarial testing (red-teaming)
\end{itemize}

\textbf{When Needed:}
\begin{itemize}
\item Complex multi-step reasoning tasks
\item High-stakes applications (legal, medical)
\item Creative content generation at scale
\item Fine-tuning training data creation
\end{itemize}


\end{column}
\begin{column}[T]{0.4\linewidth}
% \begin{center}
% \includegraphics[width=\linewidth,keepaspectratio]{prompt_engineering_example}
% \end{center}

\textbf{Salary Range:}
\begin{itemize}
\item India: Rs15-40 LPA
\item US: \$175K-\$335K
\end{itemize}

\textbf{Fun Fact:}
OpenAI job posting : "Prompt Engineer, \$300K+" went viral. Shows market value of this skill.

\vspace{0.3cm}
\textbf{Reality:}
Will likely merge into AI Engineer role long-term, but hot specialty in today's time.

\vspace{0.3cm}
\textbf{Career Note:}
Often a specialization within AI Engineer role, but some companies hire dedicated prompt engineers.
\end{column}
\end{columns}

{\tiny (Ref: OpenAI Career Page , Anthropic Hiring Posts)}
\end{frame}

\begin{frame}[fragile]\frametitle{The Modern Data Scientist: How the Role Changed}

\begin{columns}
\begin{column}[T]{0.5\linewidth}

\textbf{Old Focus (Pre-2022):}
\begin{itemize}
\item Building ML models from scratch
\item Feature engineering
\item Hyperparameter tuning
\item Model selection (RF vs XGBoost vs DL)
\item Deployment (90\% never deployed)
\end{itemize}

\textbf{Reality Check:}
Most traditional DS projects failed. Models sat unused. ROI unclear.

\end{column}
\begin{column}[T]{0.5\linewidth}

\textbf{New Focus (2024+):}
\begin{itemize}
\item \textbf{Data Curation} for fine-tuning LLMs
\item Creating evaluation datasets
\item Prompt engineering at scale
\item Small-model optimization (distillation)
\item Hallucination detection
\item RAG system design
\end{itemize}


\end{column}
\end{columns}

\begin{frame}[fragile]\frametitle{AI Safety \& Alignment Engineer (Emerging Field)}

Ensure AI systems are safe, aligned with human values


\begin{columns}
    \begin{column}[T]{0.5\linewidth}


\textbf{Core Focus Areas:}
\begin{itemize}
\item Preventing hallucinations and misinformation
\item Detecting and mitigating bias
\item Red-teaming AI systems (adversarial testing)
% \item Implementing Constitutional AI principles
\item Developing safety evaluation frameworks
\item Preventing jailbreaks and prompt injections
\end{itemize}

\textbf{Key Techniques:}
\begin{itemize}
\item RLHF (Reinforcement Learning from Human Feedback)
\item DPO (Direct Preference Optimization)
% \item Constitutional AI training
% \item Adversarial prompt testing
% \item Output verification systems
\end{itemize}
    \end{column}

\textbf{Requirements:}
