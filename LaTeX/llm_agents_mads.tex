\begin{frame}[fragile]\frametitle{History of Multi-Agent Systems}
\begin{itemize}
    \item \textbf{1950s-1970s}: Early AI research on distributed problem solving
    \item \textbf{1980s}: Distributed Artificial Intelligence (DAI) emerges
    \begin{itemize}
        \item Focus on coordinating multiple AI systems
        \item Research in distributed reasoning and planning
    \end{itemize}
    \item \textbf{1990s}: Multi-Agent Systems (MAS) becomes distinct field
    \begin{itemize}
        \item Agent communication languages (KQML, FIPA-ACL)
        \item Belief-Desire-Intention (BDI) architectures
    \end{itemize}
    \item \textbf{2000s-2010s}: Practical applications emerge
    \begin{itemize}
        \item Robotic swarms, game AI, trading systems
    \end{itemize}
    \item \textbf{2020s}: LLM revolution transforms multi-agent systems
    \begin{itemize}
        \item Natural language coordination becomes possible
        \item Agents can reason, plan, and communicate effectively
    \end{itemize}
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Why Multi-Agent Systems?}
\textbf{Limitations of Single Agent Systems}:
\begin{itemize}
    \item Limited expertise scope - cannot excel at everything
    \item Single point of failure - no redundancy
    \item Bounded cognitive capacity - context window limits
    \item Lack of diverse perspectives - prone to bias
\end{itemize}

\vspace{0.3cm}
\textbf{Advantages of Multi-Agent Systems}:
\begin{itemize}
    \item \textbf{Specialization}: Each agent masters specific domain
    \item \textbf{Parallelization}: Simultaneous processing of subtasks
    \item \textbf{Robustness}: System continues if one agent fails
    \item \textbf{Scalability}: Add agents for new capabilities
    \item \textbf{Modularity}: Easy to update or replace individual agents
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Multi-Agent System Architectures: Overview}
\textbf{Three Main Architectural Patterns}:

\begin{enumerate}
    \item \textbf{Hierarchical/Manager-Worker}
    \begin{itemize}
        \item Central coordinator delegates to specialist agents
        \item Clear authority structure, predictable flow
    \end{itemize}
    
    \item \textbf{Peer-to-Peer/Collaborative}
    \begin{itemize}
        \item Agents communicate directly, negotiate solutions
        \item Democratic decision-making, emergent behavior
    \end{itemize}
    
    \item \textbf{Pipeline/Sequential}
    \begin{itemize}
        \item Each agent performs stage in workflow
        \item Output of one becomes input of next
    \end{itemize}
\end{enumerate}
\end{frame}

    \begin{itemize}
        \item Manager creates overall plan, assigns tasks
        \item Example: MADS Oracle routing to specialists
    \end{itemize}
    
    \item \textbf{Distributed Planning}
    \begin{itemize}
        \item Each agent plans independently, negotiates conflicts
        \item Example: Market-based task allocation
    \end{itemize}
    
    \item \textbf{Reactive Coordination}
    \begin{itemize}
        \item Agents respond to environment and each other
        \item Example: Swarm robotics, stigmergy
    \end{itemize}
    
    \item \textbf{Contract Net Protocol}
    \begin{itemize}
        \item Agents bid for tasks, coordinator assigns to best bidder
        \item Example: Task allocation in heterogeneous systems
    \end{itemize}
\end{itemize}

\begin{frame}[fragile]\frametitle{Challenges in Multi-Agent Systems}
\begin{itemize}
    \item \textbf{Coordination Overhead}
    \begin{itemize}
        \item More agents = more communication = higher latency and cost
        \item Need efficient routing and synthesis mechanisms
    \end{itemize}
    
    \item \textbf{Consistency and Conflict Resolution}
    \begin{itemize}
        \item Agents may produce contradictory recommendations
        \item Need robust conflict resolution and voting mechanisms
    \end{itemize}
    
    \item \textbf{Non-Determinism}
    \begin{itemize}
        \item LLM variability multiplied across agents
        \item Difficult to guarantee reproducible results
    \end{itemize}
    
    \item \textbf{Debugging and Observability}
    \begin{itemize}
        \item Hard to trace decisions through multi-agent interactions
        \item Need comprehensive logging and visualization tools
    \end{itemize}
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Best Practices for Multi-Agent Systems}
\begin{itemize}
    \item \textbf{Start Simple}: Begin with 2-3 agents, scale gradually
    \item \textbf{Clear Roles}: Define distinct, non-overlapping agent responsibilities
    \item \textbf{Explicit Communication}: Use structured formats (JSON, XML)
    \item \textbf{State Management}: Maintain shared context effectively
    \item \textbf{Termination Conditions}: Prevent infinite loops with clear stopping criteria
    \item \textbf{Human-in-Loop}: Include human oversight for critical decisions
    \item \textbf{Evaluation Metrics}: Define success criteria for system performance
    \item \textbf{Error Handling}: Graceful degradation when agents fail
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Three-Stage Intelligence Architecture}
\begin{center}
\textbf{Oracle (Classifier) → Specialist Agents (Analyzers) → Synthesizer (Integrator)}
\end{center}

\begin{enumerate}
    \item \textbf{Oracle Agent}: Analyzes problem, routes to 3-7 most relevant specialists
    \item \textbf{Specialist Agents}: Each applies unique framework in parallel
    \item \textbf{Synthesizer Agent}: Evaluates, ranks, integrates into unified recommendation
\end{enumerate}

\vspace{0.3cm}
This three-agent core enables:
\begin{itemize}
    \item Intelligent problem classification
    \item Parallel expert analysis
    \item Coherent strategy synthesis
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{How Problems Flow Through MADS}
\begin{enumerate}
    \item \textbf{Intake}: User describes strategic situation in natural language
    \item \textbf{Classification}: Oracle extracts key dimensions (players, timing, uncertainty)
    \item \textbf{Routing}: Oracle selects most relevant 3-7 specialist agents
    \item \textbf{Parallel Analysis}: Selected agents simultaneously analyze from unique frameworks
    \item \textbf{Evaluation}: Synthesizer scores each output on multiple criteria
    \item \textbf{Integration}: Synthesizer resolves conflicts, finds synergies
    \item \textbf{Output}: Executive summary + strategic guidance + implementation plan
\end{enumerate}
\end{frame}
