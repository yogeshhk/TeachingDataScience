\begin{frame}[fragile] \frametitle{What is AI-ML-DL?}
\begin{itemize}
\item Artificial Intelligence: mimicking human intelligence
\item Machine Learning: automating learning with human-designed features -- essentially optimizing weights to best make a final prediction
\item There could be programmed (hand-coded) AI that's not Machine Learning
\item Machine Learning could be used for non-AI activities too, like automation
\item Deep Learning: neural networks that learn their own features, with no hand-designed input features
\end{itemize}
\end{frame}

\begin{frame}[fragile] \frametitle{History}
\begin{itemize}
\item 1958 -- Perceptron unit, Frank Rosenblatt
\item 1986 -- Backpropagation, Geoffrey Hinton
\item 1989 -- LeNet: backpropagation applied to multi-layer perceptrons, Yann LeCun
\item 1997 -- LSTM, Sepp Hochreiter and J\"urgen Schmidhuber
\item 2009 -- GPUs for deep learning, Andrew Ng
\item 2012 -- AlexNet wins ImageNet, error drops from 25\% to 16\%
\item 2014 -- Generative Adversarial Networks, Ian Goodfellow \& Yoshua Bengio
\item 2016 -- AlphaGo beats world champions, Google DeepMind
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile] \frametitle{Use Deep Learning When \ldots}
	\begin{itemize}
	\item You have lots of data (about 10k+ examples)
	\item The problem is ``complex'' -- speech, vision, natural language
	\item The data is unstructured
	\item You need the absolute ``best'' model
	\end{itemize}
	\begin{itemize}
	\item[] \textbf{Don't}, when: the dataset is small, traditional ML already performs well, the data is structured with known domain knowledge, or the model needs to be explainable.
	\end{itemize}
\tiny{(Ref: Introduction to TensorFlow 2.0 - Brad Miro)}
\end{frame}

\begin{frame}[fragile] \frametitle{Deep Learning == Neural Nets}
\begin{itemize}
\item Main idea: transform the input space into outputs via higher-level abstractions.
\item Neural net architectures are made up of perceptrons (similar to neurons).
\item Each neuron carries certain transformations on the inputs coming to it.
\item A collection of such neurons, with various types of transformations, can create the desired overall transformation.
\end{itemize}
\begin{center}
\includegraphics[width=0.35\linewidth,keepaspectratio]{neurons.jpg}

\end{center}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile] \frametitle{Inspiration: From Biology to Artificial Neuron}
\begin{itemize}
\item Attempts to simulate biological neural systems.
\item The human brain has approximately $10^{11}$ neurons, each connected on average to 10,000 others -- about $10^{15}$ connections in total.
\item An artificial neuron contains a nonlinear activation function and has several incoming and outgoing weighted connections.
\end{itemize}

\includegraphics[width=0.5\linewidth,keepaspectratio]{brain_neuron}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile] \frametitle{Building the Neuron: Weighted Sum + Activation}
	\begin{itemize}
	\item Idea 1: sum of the inputs? Does not work -- no way to weigh inputs differently.
	\item Idea 2: weight each input differently -- gains expressivity. Remind you of anything (linear regression)?
	\item Idea 3: transform the weighted sum via a (nonlinear) activation function.
	\end{itemize}

\begin{frame}[fragile] \frametitle{Why Deep?}
\begin{itemize}
\item Many-representation, multi-level graph architectures -- cannot be done with shallow models like SVM, Naive Bayes, etc.
\item The human brain learns language, speech, and images through a sequence of processing areas.
\item Humans organize ideas and concepts hierarchically, learning simpler concepts first and composing them into more abstract ones.
\item Engineers similarly break solutions up into multiple levels of abstraction and processing.
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Usage Requirements}
\begin{itemize}
\item Large data set with good quality (input-output mappings)
\item Measurable and describable goals (define the cost)
\item Enough computing power (AWS GPU Instance)
\item Excels in tasks where the basic unit (pixel, word) has very little meaning in itself, but the combination of such units has a useful meaning.
