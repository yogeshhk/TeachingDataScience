\begin{frame}[fragile]\frametitle{What is a Language Model?}
\begin{itemize}
\item While typing SMS, have you seen a next-word suggestion appear?
\item That is a \textbf{language model}: it assigns a probability to each possible next word given the words seen so far.
\item Think of it as: given everything said so far, what word is most likely to come next?

\item \textbf{Generative model}: keep sampling the next token until a complete response is formed.

\begin{frame}[fragile]\frametitle{What Is a Model? Parameters \& Knobs}
\begin{itemize}
\item A model replaces a lookup table with a compact mathematical function.
\item Example: a line $y = ax + b$ needs only 2 parameters to describe infinitely many $(x,y)$ pairs.
\item GPT-3 uses \textbf{175 billion parameters} – enough ``knobs'' to approximate the distribution of human language.
\item Training adjusts these parameters so the model assigns high probability to the actual next token in the training data.
\end{itemize}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Tokens: The Unit LLMs Operate On}
\begin{itemize}
\item LLMs don't operate on words but on \textbf{tokens} (sub-word units).
\item Tokeniser (e.g.\ BPE, WordPiece) breaks text into pieces:
  \begin{center}
  \texttt{"playing"} $\rightarrow$ \texttt{["play", "\#\#ing"]}\\
  \texttt{"unbelievable"} $\rightarrow$ \texttt{["un", "\#\#believ", "\#\#able"]}
  \end{center}
\item Benefits: fixed vocabulary ($\sim$30K--50K tokens), handles rare/new words gracefully.
\item GPT-3: trained on 499B tokens; GPT-4: training data size undisclosed by OpenAI; estimates suggest well over 1T tokens.
\item In comparison: a human reading 12 hr/day for 80 years absorbs $\sim$5.3B words – a tiny fraction of GPT-3's training data alone.
\end{itemize}
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Next-Word Prediction}
\begin{itemize}
\item For the text \texttt{"The best thing about AI is its ability to"}, what comes next?
\item A language model produces a \textbf{ranked probability distribution} over all vocabulary words.
\item Sampling strategy matters:
  \begin{itemize}
  \item \textbf{Greedy (temp=0)}: always pick highest-prob word $\rightarrow$ flat, repetitive text.
  \item \textbf{Temperature 0.8}: occasionally pick lower-prob words $\rightarrow$ varied, creative output.
  \end{itemize}
\end{itemize}
\begin{center}
\includegraphics[width=0.6\linewidth,keepaspectratio]{llm1}
\end{center}
{\tiny (Ref: What Is ChatGPT Doing\ldots – S.\ Wolfram)}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Sampling Strategies at Inference}
\begin{itemize}
\item \textbf{Greedy decoding}: always pick $\arg\max P(w)$ – fast, but repetitive and deterministic.
\item \textbf{Top-$k$ sampling}: sample from the $k$ most probable tokens (e.g.\ $k=50$).
\item \textbf{Top-$p$ (nucleus) sampling}: sample from the smallest set of tokens whose cumulative probability $\ge p$ (e.g.\ $p=0.9$).
\item \textbf{Temperature}: divide logits by $T$ before softmax.
  \begin{itemize}
  \item $T \to 0$: near-greedy; $T=1$: standard; $T > 1$: more random/creative.
  \end{itemize}
\item \textbf{Beam search}: maintain top-$b$ partial sequences; pick the one with highest joint probability.
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Word Vectors: Representing Words as Numbers}
\begin{columns}
  \begin{column}[T]{0.55\linewidth}
    \begin{itemize}
    \item Each word $\rightarrow$ a dense vector of $d$ real numbers (e.g.\ $d=300$).
    \item Geometrically similar words cluster together.
    \item Famous property: $\text{King} - \text{Man} + \text{Woman} \approx \text{Queen}$.
    \item Problem: static vectors – "bank" (river) and "bank" (financial) share one vector.
    \end{itemize}
  \end{column}
  \begin{column}[T]{0.45\linewidth}
    \begin{center}

    \end{center}
  \end{column}
\end{columns}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Seq2Seq: Encoder-Decoder with RNN}
\begin{center}
\includegraphics[width=0.8\linewidth,keepaspectratio]{nlp14}
\end{center}
\begin{itemize}
\item \textbf{Encoder} reads the input sequence, compresses it into a fixed-size \emph{context vector}.
\item \textbf{Decoder} generates the output word-by-word, conditioned on that context vector.
\item \textbf{Problem}: single context vector is an information bottleneck for long sequences.
\end{itemize}
{\tiny (Ref: Deep Learning and NLP A-Z – K.\ Eremenko)}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Attention Mechanism: Breaking the Bottleneck}
\begin{center}
\includegraphics[width=0.8\linewidth,keepaspectratio]{llm118}
\end{center}
\begin{itemize}
\item Instead of one context vector, the decoder \textbf{attends} to all encoder hidden states.
\item Each decoder step computes a weighted sum over encoder states,  the weight reflects how relevant each input word is right now.
\item Think of it like a spotlight: when generating the word ``bank'' in French, the model shines the spotlight back on the word ``bank'' in English.
\end{itemize}
{\tiny (Ref: CS447 Natural Language Processing – J.\ Hockenmaier)}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Contextualized Embedding}
\begin{itemize}
\item \textbf{The problem with static embeddings}: Word2Vec assigns one
  fixed vector to each word regardless of context. The word
  \texttt{Apple} gets a single vector, an average of its ``fruit''
  and ``company'' meanings. That average is accurate for neither.
\item \textbf{Contextualised embeddings fix this}: instead of a fixed
  vector, each occurrence of \texttt{Apple} gets a \emph{different}
  vector shaped by the words around it. ``I love Apple Juice'' pulls
  the vector toward fruit; ``Apple posted record revenue'' pulls it
  toward technology company.
\item ``You shall know a word by the company it keeps.'' – J.\ R.\ Firth
\item Transformers achieve this through \textbf{self-attention}: each token's
  representation is updated by a weighted sum of all other tokens' \emph{Value}
  vectors. The weights come from the dot product of that token's \emph{Query}
  with every other token's \emph{Key}, so semantically related tokens
  contribute more.
\end{itemize}

\end{frame}

\begin{frame}[fragile]\frametitle{Self-Attention: Every Token Attends to Every Token}
\begin{itemize}
\item \textbf{Cross-attention} (seq2seq): decoder attends to \emph{encoder} positions.
\item \textbf{Self-attention}: every position in a sequence attends to every \emph{other position in the same sequence}.
\item Enables the encoder to build rich, contextualised representations \emph{without} recurrence.
\item Enables full \textbf{parallelisation}: all tokens processed simultaneously during training,  dramatically faster than RNNs.
\end{itemize}
\vspace{0.5em}
Self-attention is the core innovation of the Transformer, replacing sequential RNN processing with parallel, content-based lookups.
\end{frame}

\begin{frame}[fragile]\frametitle{Step 2: Positional Encoding, The Problem}
\begin{itemize}
\item The Transformer processes all tokens in parallel, its key
  advantage over RNNs. But parallelism means token order is no
  longer implicit.
\item Unlike RNNs, which encode order implicitly through sequential hidden-state
  updates, the Transformer's self-attention is \emph{permutation-equivariant}:
  shuffle the input tokens and the attention outputs shuffle identically.
  We must inject position information explicitly.
\item A raw integer index (0, 1, 2\ldots) is unsuitable: large values at long
  sequences would dominate the embedding and generalise poorly to lengths
  not seen in training.
\item Normalising indices to $[0,1]$ also fails: the step size
  between positions depends on sequence length, so the model
  cannot learn a consistent meaning of ``adjacent.''
\item \textbf{Requirements for a good positional signal}: bounded
  magnitude, unique per position, consistent across sequence lengths,
  and preservable by the model's arithmetic operations.
\end{itemize}


\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Which function(s) to use?}
\begin{itemize}
\item If you use single sine function, ie $sine(i)$, then at multiple places you have same value (as it is periodical). So instead of single scalar value from one curve, overlay it with say cosine function on alternate tokens. But still, at some places even their addition can repeat. Then add again sine, cosine pairs with different frequencies. Till $L/2$ pairs. Like a Fourier transform. Summation is unique for each token.
\item The original Transformer uses sine/cosine waves at different frequencies,  each position gets a unique pattern/vector of numbers added to its embedding.
\item Think of it as a \textbf{unique address tag} for each position in the sentence.
\item  Note that even positions correspond to a sine function and odd positions correspond to cosine functions.
\item Added element-wise to the token embedding:
\[
  \mathbf{x}_i = \mathbf{e}_i + \text{PE}(i)
\]
\end{itemize}


\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Positional Encoding Formula}

\begin{itemize}

\item Suppose you have an input sequence of length $L$ and require the position of the $k$th object within this sequence. The positional encoding is given by sine and cosine functions of varying frequencies::
\[
  \text{PE}(k, 2i)   = \sin\!\left(\frac{k}{n^{2i/d}}\right), \quad
  \text{PE}(k, 2i+1) = \cos\!\left(\frac{k}{n^{2i/d}}\right)
\]
\item $k$: position index of a token in the sequence, $0 \leq k \leq L-1$
\item $d$ : Dimension of the output embedding space
\item $i$: dimension pair index, $0 \leq i \leq d/2 - 1$, with a single value of $i$ mapping to both sine and cosine functions
\item $n$: User-defined scalar, set to 10000 by the author

\end{itemize}

\begin{frame}[fragile]\frametitle{Positional Encoding Explanation}

\begin{center}
\includegraphics[width=0.7\linewidth,keepaspectratio]{llm253}
\end{center}

\textit{Plain English}: position $k$ in the sequence gets a vector of
$d$ numbers. Even-indexed dimensions use a sine wave and odd-indexed
dimensions use a cosine wave, each at a different frequency controlled
by $n^{2i/d}$. Because the frequencies differ, no two positions produce
the same vector, every position has a unique ``fingerprint'' that the
model can read.

\begin{frame}[fragile]\frametitle{Positional Encoding: Why This Method?}
\begin{itemize}
\item \textbf{Why not just append the position number?}
  A raw integer like $42$ has no bounded magnitude – it would dominate the embedding
  and scale differently at test-time lengths never seen in training.
\item \textbf{Why sinusoids?} Three properties make them ideal:
  \begin{enumerate}
  \item \textbf{Bounded}: $\sin/\cos \in [-1, 1]$ – same scale as normalised embeddings.
  \item \textbf{Unique per position}: each position gets a distinct vector of
        $(\sin, \cos)$ values across frequencies.
  \item \textbf{Relative positions are learnable}: $\text{PE}(pos+k)$ is a
        \emph{linear function} of $\text{PE}(pos)$, so the model can learn
        to attend by relative offset, not just absolute position.
  \end{enumerate}
\item \textbf{Why add, not concatenate?}
  Addition keeps the vector dimension $d$ fixed.
  Concatenation would double $d$, doubling all subsequent weight matrices.
  Empirically, addition works just as well and is cheaper.
\item \textbf{Modern alternative – RoPE} (Rotary Position Embedding):
  encodes position directly into the $Q$–$K$ dot product so attention scores
  are naturally relative; used in LLaMA, Mistral, GPT-NeoX.
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Step 3: Scaled Dot-Product Attention}
\begin{itemize}
\item $W^Q$, $W^K$, $W^V$ are \textbf{learned weight matrices},
  one set per attention head, updated during training. They project
  the same input token vector $\mathbf{X}$ into three different
  ``views'' of that token, each suited to a different role in the
  attention computation.
\item For each token, compute three projections:
\[
  \mathbf{Q} = \mathbf{X}W^Q, \quad \mathbf{K} = \mathbf{X}W^K, \quad \mathbf{V} = \mathbf{X}W^V
\]
\item Attention formula:
\[
  \text{Attention}(\mathbf{Q}, \mathbf{K}, \mathbf{V}) 
    = \text{softmax}\!\left(\frac{\mathbf{Q}\mathbf{K}^\top}{\sqrt{d_k}}\right)\mathbf{V}
\]
\item \textbf{In plain English}: compute how relevant each word is to every other word (dot product), normalise to get weights (softmax), then mix the information accordingly.
\item The $\sqrt{d_k}$ scaling prevents scores from getting too large and making the model overconfident.
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Q, K, V: Meaning and Intuition}
\textbf{Think of attention as a soft database lookup:}
\begin{center}
\small
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|l|l|p{5.5cm}|}
\hline
\textbf{Vector} & \textbf{Analogy} & \textbf{What it represents} \\
\hline
\textbf{Query} $Q_i$ & Search query & ``What information does token $i$ need?'' \\
\hline
\textbf{Key}   $K_j$ & Index tag    & ``What information does token $j$ offer?'' \\
\hline
\textbf{Value} $V_j$ & Actual data  & ``The content token $j$ will contribute if selected'' \\
\hline
\end{tabular}
\end{center}
\vspace{0.3em}
\textbf{Word-level example} for sentence \texttt{"The cat sat on the mat"}:
\begin{itemize}
\item Token \texttt{"sat"} issues a Query: \textit{``I'm a verb – who is my subject?''}
\item Token \texttt{"cat"} has a Key that scores highly against that Query.
\item The weighted Value of \texttt{"cat"} flows into the representation of \texttt{"sat"}.
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Q, K, V: Step-by-Step for One Word}
\textbf{Input sentence}: \texttt{"The bank can guarantee deposits"}\\[0.4em]
For the word \texttt{"bank"} (position $i = 2$):
\begin{enumerate}
\item \textbf{Project to Q, K, V}:
      $\mathbf{q}_2 = \mathbf{x}_2 W^Q,\quad
       \mathbf{k}_2 = \mathbf{x}_2 W^K,\quad
       \mathbf{v}_2 = \mathbf{x}_2 W^V$
\item \textbf{Score against every other token's Key}:
      $s_{2j} = \mathbf{q}_2 \cdot \mathbf{k}_j / \sqrt{d_k}$
      for $j = 1 \ldots T$
\item \textbf{Normalise scores}: $\alpha_{2j} = \text{softmax}(s_{2j})$
      \hfill {\small (sum to 1 across $j$)}
\item \textbf{Aggregate Values}:
      $\mathbf{z}_2 = \sum_j \alpha_{2j}\,\mathbf{v}_j$
\end{enumerate}
\vspace{0.3em}
After seeing ``deposits'' and ``guarantee'', attention weights for \texttt{"bank"} should
be high on those tokens $\Rightarrow$ \texttt{"bank"} gets a \emph{financial} representation.
In a sentence about a river, weights shift to ``river'', ``flow'' $\Rightarrow$
\texttt{"bank"} gets a \emph{geographical} representation.\\[0.3em]
This is how the Transformer achieves \textbf{contextualised embeddings}.
\end{frame}

\begin{frame}[fragile]\frametitle{Step 4: Multi-Head Attention}
\begin{itemize}
\item Not just a single word may mean different things (e.g. Apple or Light etc), but even a single full sentence can mean different things.
\item Example: ``She saw the man with the telescope''. It has two meanings.
\item Multiple self-attention layers capture different sentence/meaning patterns simultaneously, each head attending to a different type of linguistic relationship.
\end{itemize}
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 4: Multi-Head Attention}
\begin{itemize}
\item Run $h$ attention functions \textbf{in parallel}, each with independent $W^Q_l, W^K_l, W^V_l$ projections:
\[
  \text{MultiHead}(\mathbf{X}) = \text{Concat}(\text{head}_1, \ldots, \text{head}_h)\,W^O
\]
\item \textbf{Why multiple heads?} Each head specialises in a different type of relationship:
  \begin{itemize}
  \item Head 1: syntactic agreement (subject–verb)
  \item Head 2: co-reference (pronoun resolution)
  \item Head 3: positional proximity … and so on
  \end{itemize}
\item Think of it like multiple people reviewing the same sentence simultaneously, each looking for something different,  then combining their findings.
\item BERT-base uses $h=12$ heads; GPT-3 uses $h=96$.
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Step 5: Add \& Norm}
\textbf{Residual Connection + Layer Norm:}
\[
  \mathbf{h} = \text{LayerNorm}(\mathbf{x} + \text{MultiHead}(\mathbf{x}))
\]
\begin{itemize}
\item \textbf{Why gradients vanish}: during training, error signals
  flow backward through every layer via multiplication. In deep
  networks, repeatedly multiplying small numbers drives the signal
  toward zero, early layers receive almost no learning signal.
  The residual (skip) connection provides a direct addition path,
  so the gradient can travel straight to early layers without
  passing through every multiplication.
\item \textbf{Residual (skip) connection}: allows gradients to flow directly to earlier layers, preventing vanishing gradients.
\item \textbf{Layer Norm}: normalises across features for each token independently, stabilising training.
\end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 5:  Feed-Forward Network}

\textbf{Position-wise Feed-Forward Network (FFN):}
\begin{itemize}
\item Applied independently to \emph{each token} after attention.
\item Attention \emph{collects} information from other tokens; FFN \emph{processes} it,  adding non-linearity and depth to each token's representation.
\item Think of attention as gathering evidence, and FFN as reasoning about it.
\end{itemize}
\end{frame}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{One Full Encoder Pass: Putting It Together}
For input sentence \texttt{``The bank can guarantee deposits''}:
\begin{enumerate}
  \item \textbf{Tokenise}: split into sub-word tokens via BPE.
  \item \textbf{Embed}: each token ID $\rightarrow$ dense vector $\mathbf{e}_i$.
  \item \textbf{Add positional encoding}: $\mathbf{x}_i = \mathbf{e}_i + \text{PE}(i)$.
  \item \textbf{Multi-head self-attention}: each token attends to all others;
    ``bank'' shifts toward ``guarantee''/``deposits''.
  \item \textbf{Add \& Norm}: residual connection stabilises gradients.
  \item \textbf{Feed-forward network}: each token representation processed
    independently.
  \item \textbf{Repeat} steps 4--6 for $N$ layers (e.g.\ $N=12$ for
    BERT-base).
  \item \textbf{Output}: contextualised vector per token, ready for a
    task head.
\end{enumerate}
\textit{Plain English: each layer refines what each token ``means'' given
its neighbours, stacking understanding layer by layer.}
\end{frame}

\begin{frame}[fragile]\frametitle{GPT Inference: What Happens After You Send a Prompt}
\textbf{Example prompt}: \texttt{"Translate to French: The sky is"}

\begin{enumerate}
\item \textbf{Tokenise}: prompt split into token IDs by the BPE tokeniser.
\item \textbf{Embed + positional encoding}: each ID $\rightarrow$ embedding vector; add positional tag.
\item \textbf{Forward pass through all $N$ decoder layers}:
      causal self-attention + FFN, repeated $N$ times (e.g.\ $N=96$ for GPT-3).
\item \textbf{Project to vocabulary}: final hidden state $\rightarrow$ logit vector of size $\sim$50K.
\item \textbf{Sample}: apply temperature + softmax; sample (or argmax) one token, e.g.\ \texttt{"bleu"}.
\item \textbf{Append \& repeat}: \texttt{"bleu"} is appended to the sequence; repeat until \texttt{<|endoftext|>}.
\end{enumerate}
\vspace{0.3em}
\textbf{Key-Value Cache}: Keys and Values for all previous tokens are cached.  Only the new token's Q, K, V are computed each step,  making autoregressive generation efficient.
\end{frame}

\begin{frame}[fragile]\frametitle{Key Stages in LLM Development}
\begin{enumerate}
\item \textbf{Pre-training}: self-supervised on massive raw text corpus.\\
  {\small High cost, long duration, high expertise required.}
\item \textbf{Prompt Engineering}: craft effective prompts to steer the pre-trained model.\\
  {\small No training cost; limited customisation.}
\item \textbf{Supervised Fine-Tuning (SFT)}: train on small curated (prompt, response) pairs.\\
  {\small Low–medium cost; teaches instruction-following.}
\item \textbf{Reinforcement Learning from Human Feedback (RLHF)}: align model to human values.\\
  {\small Medium–high cost; requires reward model.}
\item \textbf{Deployment}: quantisation, distillation, caching optimisations.\\
  {\small Reduces inference cost and latency.}
\end{enumerate}
\end{frame}

\begin{frame}[fragile]\frametitle{Supervised Fine-Tuning (SFT): What Actually Happens}
\begin{itemize}
  \item The pre-trained model knows language but not how to
    \emph{follow instructions}. SFT teaches it that.
  \item Training data: human-written (prompt, ideal response) pairs
   , typically thousands to tens of thousands of examples.
  \item The model is trained to maximise the probability of the ideal
    response given the prompt, using the same cross-entropy loss as
    pre-training.
  \item Because the dataset is small relative to pre-training,
    \textbf{LoRA} (Low-Rank Adaptation) is standard: freeze most
    weights, train only small adapter matrices.
    \begin{itemize}
      \item Reduces trainable parameters from billions to millions.
      \item Fits on a single consumer GPU.
    \end{itemize}
  \item Output: an \emph{instruction-following} model (e.g.\
    \texttt{llama-3-8b-instruct}).
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Retrieval-Augmented Generation (RAG)}

    \begin{itemize}
    \item LLMs have static knowledge frozen at training time.
    \item \textbf{RAG} combines a retrieval system with an LLM generator.
    \item Steps:
      \begin{enumerate}
      \item Embed query with an encoder model.
      \item Search a \textbf{vector database} (Chroma, Pinecone, Weaviate) for relevant chunks.
      \item Inject retrieved context into the prompt.
      \item Decoder-LLM generates a grounded answer.
      \end{enumerate}
    \item Reduces hallucinations; enables up-to-date information.
    \end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{LLM Limitations: What They Cannot Do}
\begin{itemize}
\item \textbf{No beliefs or knowledge}: LLMs are next-token predictors; they have no semantic understanding in the human sense.
\item \textbf{Hallucination}: will confidently generate plausible-sounding but factually wrong content.
\item \textbf{Autoregressive error accumulation}: a wrong early token propagates; no backtracking or self-correction.
\item \textbf{Context window limit}: conversations appear coherent until the log exceeds the context length; earlier parts are "forgotten."
\item \textbf{Static knowledge}: training has a cutoff date; model cannot access real-time information (unless augmented with RAG/tools).
\item \textbf{Uncertain reasoning ability}: standard LLMs produce
  plan-like outputs through pattern matching, not verified look-ahead.
  Newer reasoning models (OpenAI o1, DeepSeek-R1) extend
  chain-of-thought at inference time and show stronger multi-step
  problem solving, but whether this constitutes ``true reasoning''
  remains an active research question.
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Ethical Considerations}
\begin{itemize}
\item LLMs can generate racist, sexist, or harmful content, they reflect the biases in their training data.
\item Awareness and continuous monitoring of ethical implications is essential.
\item \textbf{Principles for responsible use}:
  \begin{itemize}
  \item Transparency: disclose AI involvement.
  \item Accountability: humans remain responsible for LLM outputs.
  \item Fairness: test for demographic disparities.
  \item Privacy: do not feed sensitive personal data into public APIs.
  \end{itemize}
\item \textbf{Always verify}: never blindly trust LLM output in high-stakes domains (medical, legal, financial, safety-critical).
\end{itemize}
