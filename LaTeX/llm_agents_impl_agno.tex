\begin{frame}[fragile]\frametitle{Introduction to Agno Framework}
      \begin{itemize}
	\item \textbf{Agno} (formerly Phidata) is an open-source, lightweight framework for building Multi-Agent Systems with memory, knowledge, and reasoning
	\item Official documentation: \url{https://docs.agno.com/examples/introduction}
	\item Designed for \textbf{speed and efficiency}; build production-ready agentic systems with minimal boilerplate
	\item Exposes LLMs as a unified API and adds superpowers: memory, knowledge, tools, and reasoning
	\item Supports \textbf{5 progressive levels} of agentic system complexity
	\item Built for both individual agents and complex multi-agent workflows
	  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% SLIDE 136 — fixed formatting: stray newline in bullet text removed
\begin{frame}[fragile]\frametitle{Key Components}
    \begin{itemize}
	\item \textbf{Agents:} Autonomous reasoning units with configurable goals and instructions
	\item \textbf{Tools:} Plugins that give agents extraordinary abilities (search, finance, code execution, etc.)
	\item \textbf{Memory and Knowledge:} Agents remember past interactions and store relevant information, enabling context across conversations
	\item \textbf{Knowledge Stores:} Connect to Vector databases for Retrieval-Augmented Generation (RAG)
	\item \textbf{Multi-Agent Orchestration:} Create teams of agents that collaborate on complex tasks
	  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{The 5 Levels of Agentic Systems}
      \begin{itemize}
	\item \textbf{Level 1:} Agents with tools and instructions
	      \\ \textit{Basic autonomous task execution, the ``Hello World'' of agents}
	\item \textbf{Level 2:} Agents with knowledge and storage
	      \\ \textit{Persistent data management; agent remembers across sessions}
	\item \textbf{Level 3:} Agents with memory and reasoning
	      \\ \textit{Context-aware decision-making and chain-of-thought planning}
	\item \textbf{Level 4:} Agent Teams that reason and collaborate
	      \\ \textit{Multi-agent coordination with shared goals}
	\item \textbf{Level 5:} Agentic Workflows with state and determinism
	      \\ \textit{Complex orchestrated processes; business-grade reliability}
	\item Each level \textbf{builds on} the previous, enabling progressively sophisticated systems
	  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Key Features, Model Support \& Performance}
      \begin{itemize}
	\item \textbf{Model Agnostic:} Unified interface to 23+ model providers, no vendor lock-in
	\item \textbf{High Performance:} Agents instantiate in approximately \textbf{3 microseconds}
	\item \textbf{Memory Efficient:} Uses only \textbf{6.5\,KB} memory on average per agent
	\item \textbf{Scalable Architecture:} Designed for production workloads and high throughput
	\item \textbf{Security:} AgentOS runs in your own cloud; no data leaves your system
	\item \textbf{Production Ready:} Built-in FastAPI routes for immediate deployment
	  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Level 1 Agent Example, Basic Setup}
      \begin{itemize}
	\item Basic reasoning agent with \textbf{YFinance} API integration
	\item Uses \textbf{Claude Sonnet 4} for advanced reasoning capabilities
	\item \texttt{ReasoningTools} adds a structured think-before-act step
	\item \texttt{YFinanceTools} provide live stock market data access
	  \end{itemize}
      
      \begin{lstlisting}[language=python, basicstyle=\tiny]
from agno.agent import Agent
from agno.models.anthropic import Claude
from agno.tools.reasoning import ReasoningTools
from agno.tools.yfinance import YFinanceTools

agent = Agent(
    model=Claude(id="claude-sonnet-4-20250514"),
    tools=[
        ReasoningTools(add_instructions=True),
        YFinanceTools(stock_price=True,
                     analyst_recommendations=True,
                     company_info=True, company_news=True),
    ],
    instructions=["Use tables to display data",
                  "Only output the report, no other text"],
    markdown=True,
)
agent.print_response(
    "How did NVIDIA stock perform in the last two years?",
    stream=True, show_full_reasoning=True,
    stream_intermediate_steps=True)
      \end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Level 2 Agent, MCP + Persistent Storage}
      \begin{itemize}
	\item Connect the agent to Agno's documentation via an \textbf{MCP server}
	\item Give it a \textbf{SQLite database} to store conversation history and state
	\item The \textbf{AgentOS} exposes ready-to-use FastAPI endpoints for serving, monitoring, and managing agents
	  \end{itemize}
      
      \begin{lstlisting}[language=python, basicstyle=\tiny]
from agno.agent import Agent
from agno.db.sqlite import SqliteDb
from agno.models.anthropic import Claude
from agno.os import AgentOS
from agno.tools.mcp import MCPTools

agno_agent = Agent(
    name="Agno Agent",
    model=Claude(id="claude-sonnet-4-20250514"),
    db=SqliteDb(db_file="agno.db"),           # persistent storage
    tools=[MCPTools(                           # MCP tool integration
        transport="streamable-http",
        url="https://docs.agno.com/mcp")],
    add_history_to_context=True,               # short-term memory
    markdown=True,
)
agent_os = AgentOS(agents=[agno_agent])
app = agent_os.get_app()
# Run: fastapi dev agno_agent.py  -> http://localhost:8000
      \end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Connect your AgentOS}
      \begin{itemize}
	\item Agno provides a web interface that connects to your AgentOS for monitoring, management, and testing
	\item Open \texttt{os.agno.com} and sign in to your account
	\item Click \textbf{``Add new OS''} in the top navigation bar
	\item Select \textbf{``Local''} to connect to a local AgentOS on your machine
	\item Enter the AgentOS endpoint URL (default: \texttt{http://localhost:8000})
	\item Give your AgentOS a descriptive name (e.g., ``Development OS'')
	\item Click \textbf{``Connect''}
	  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Multi-Agent Teams, Architecture Principles}
      \begin{itemize}
	\item \textbf{Atomic Agents:} Individual agents work best with narrow scope and limited tools
	\item \textbf{Specialisation:} Each agent focuses on one domain of expertise
	\item \textbf{Load Distribution:} Teams spread cognitive load across multiple agents
	\item \textbf{Tool Management:} Prevent tool overload by distributing capabilities across agents
	\item \textbf{Coordinated Execution:} Team-level orchestration handles complex, multi-step tasks
	  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Multi-Agent Team, Defining Specialist Agents}
      \begin{itemize}
	\item \textbf{Web Agent}: specialises in information search and sourcing
	\item \textbf{Finance Agent}: specialises in financial data retrieval and analysis
	\item Each agent has dedicated tools and domain-specific instructions
	  \end{itemize}
      
      \begin{lstlisting}[language=python, basicstyle=\tiny]
from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.tools.duckduckgo import DuckDuckGoTools
from agno.tools.yfinance import YFinanceTools

web_agent = Agent(name="Web Agent",
    role="Search the web for information",
    model=OpenAIChat(id="gpt-4o"),
    tools=[DuckDuckGoTools()],
    instructions="Always include sources",
    show_tool_calls=True, markdown=True)

finance_agent = Agent(name="Finance Agent",
    role="Get financial data",
    model=OpenAIChat(id="gpt-4o"),
    tools=[YFinanceTools(stock_price=True,
                        analyst_recommendations=True,
                        company_info=True)],
    instructions="Use tables to display data",
    show_tool_calls=True, markdown=True)
      \end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Multi-Agent Team, Coordination \& Execution}
      \begin{itemize}
	\item \textbf{Team coordination mode} enables collaborative agent interaction
	\item Stream processing provides real-time visibility into team collaboration
	\item \texttt{pip install ddgs yfinance} for the tool dependencies
	  \end{itemize}
      
      \begin{lstlisting}[language=python, basicstyle=\tiny]
from agno.team import Team
	  
agent_team = Team(
    mode="coordinate",           # agents collaborate dynamically
    members=[web_agent, finance_agent],
    model=OpenAIChat(id="gpt-4o"),
    success_criteria=(
        "A comprehensive financial news report "
        "with clear sections and data-driven insights."),
    instructions=["Always include sources",
                 "Use tables to display data"],
    show_tool_calls=True,
    markdown=True)

agent_team.print_response(
    "What is the market outlook for AI semiconductor companies?",
    stream=True)
      \end{lstlisting}
\end{frame}
