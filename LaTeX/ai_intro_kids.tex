\begin{frame}[fragile]\frametitle{How YOU Learn New Things}
\begin{itemize}
\item Remember when you learned to ride a bicycle?
\begin{itemize}
\item First time: You fell down (made mistakes)
\item Practiced more: Got better each time
\item After many tries: You can ride without thinking!
\end{itemize}
\item Same with learning math, cricket, or dancing
\item Practice + Mistakes = Learning!
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{How Computers Learn (Machine Learning)}
\begin{itemize}
\item Computers learn the SAME WAY!
\item Example: Teaching computer to recognize cats
\begin{itemize}
\item Show 1000 pictures of cats
\item Show 1000 pictures of dogs
\item Computer finds patterns (whiskers, pointy ears, etc.)
\item Now it can tell cats from dogs in NEW pictures!
\end{itemize}
\item This is called {\bf Machine Learning} (ML)
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Three Ways Computers Learn}

{\bf 1. Supervised Learning} (Learning with a teacher)
\begin{itemize}
\item Like learning with answer key
\item Example: Showing labeled pictures - ``This is a cat'', ``This is a dog''
\end{itemize}

{\bf 2. Unsupervised Learning} (Learning by yourself)
\begin{itemize}
\item Finding patterns without help
\item Example: Grouping similar songs together on Spotify
\end{itemize}

{\bf 3. Reinforcement Learning} (Learning by trial and error)
\begin{itemize}
\item Learn by rewards and punishments
\item Example: Game AI that gets points for winning
\end{itemize}

\end{frame}

\begin{itemize}
\item Movies show robots taking over the world
\item Scary AI like in {\em Terminator}
\item Is this real?
\end{itemize}

\vspace{0.5cm}
{\bf The Truth:}
\begin{itemize}
\item Current AI is NOT like movies (yet!)
\item AI today can't ``think'' for itself
\item It only does what we train it to do
\item Like a super smart calculator, not a thinking robot
\end{itemize}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Real Problems to Think About}

{\bf Things to be careful about:}
\begin{itemize}
\item {\bf Privacy:} AI cameras everywhere - where's our privacy?
\item {\bf Fairness:} If AI is trained on biased data, it becomes biased
\begin{itemize}
\item Example: Face recognition works poorly on dark skin (not fair!)
\end{itemize}
\item {\bf Cheating:} Using ChatGPT to do ALL your homework
\begin{itemize}
\item You don't learn!
\end{itemize}
\item {\bf Fake Content:} AI can create fake videos (Deepfakes)
\begin{itemize}
\item Hard to know what's real
\end{itemize}
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{How to Use AI Responsibly}
\begin{itemize}
\item Use AI as a {\bf helper}, not a replacement for learning
\item Like using a calculator: Use it, but still learn math!
\item {\bf Good:} Ask ChatGPT to explain a concept differently
\item {\bf Not Good:} Copy-paste ChatGPT's essay for homework
\item Always {\bf verify} information from AI
\item Think critically - don't believe everything AI says!
\end{itemize}
\end{frame}
