\begin{frame}[fragile]\frametitle{How We Got Here: The Progression of AI}

\begin{itemize}
\item \textbf{Machine Learning:} Feed the computer data + labels. It learns patterns. One model per task (e.g., a spam filter only detects spam).
\item \textbf{Deep Learning:} Feed raw data (images, text) to neural networks: more powerful, still one model per task.
\item \textbf{NLP (Natural Language Processing):} Work specifically on text. Convert words to numbers (\emph{embeddings}, think of each word as a point in space), then train a model. Still one model per task.
\item \textbf{Pre-trained Models:} Train once on a huge corpus; fine-tune a small extra layer for each new task. One base model, many tasks.
\item \textbf{Large Language Models (LLMs):} Train on the internet. Just \emph{tell it what to do in plain English}. Zero-shot and few-shot: no retraining needed. One model, unlimited tasks.
\end{itemize}

\medskip
\textbf{Conclusion:} We went from writing mathematical rules, to writing code, to writing \emph{sentences} to control AI. Prompt Engineering is the natural next step.

\begin{frame}[fragile]\frametitle{Summary of Prompt Engineering}

\begin{itemize}
  \item \textbf{What is a Prompt?} The text input you give to an AI to get a response.

  \item \textbf{What is Prompt Engineering?} Crafting that input carefully so the AI reliably gives you the output you actually want.

  \item \textbf{Why does it matter?} Same AI, same task: vastly different results depending on how you ask. Better prompts mean fewer hallucinations, more reliable answers, and more powerful applications.

  \item \textbf{Core techniques (simplest first):}
    \begin{itemize}
      \item Zero-Shot and Few-Shot Prompting
      \item Chain-of-Thought for reasoning tasks
      \item Self-Consistency for high-accuracy needs
      \item RAG (\emph{Retrieval-Augmented Generation}) for private or up-to-date information
      \item Prompt Chaining for complex workflows
    \end{itemize}
\end{itemize}

\note{RAG was introduced in the Techniques section. It is the key method for grounding the AI in your own verified documents, reducing hallucination.}

\end{frame}

\begin{frame}[fragile]\frametitle{A Comprehensive Prompting Checklist}

\begin{columns}
    \begin{column}[T]{0.5\linewidth}
    \begin{itemize}
    \item \textbf{Tone}: formal, casual, persuasive?
    \item \textbf{Format}: essay, bullets, JSON, table?
    \item \textbf{Role}: expert, critic, teacher?
    \item \textbf{Objective}: inform, persuade, entertain?
    \item \textbf{Context}: background facts, data?
    \item \textbf{Scope}: how broad or narrow?
    \item \textbf{Keywords}: must-include terms?
    \item \textbf{Constraints}: word limit, topics to avoid?
    \item \textbf{Examples}: sample input/output pairs?
    \end{itemize}
    \end{column}
    \begin{column}[T]{0.5\linewidth}
    \begin{itemize}
    \item \textbf{Audience}: who will read this?
    \item \textbf{Language}: output in English, Hindi, etc.?
    \item \textbf{Citations}: should it cite sources?
    \item \textbf{Perspectives}: multiple viewpoints?
    \item \textbf{Counter arguments}: address objections?
    \item \textbf{Analogies}: use examples to clarify?
    \item \textbf{Statistics}: back claims with data?
    \item \textbf{Sensitivity}: handle certain topics carefully?
    \item \textbf{Call to action}: what should the reader do next?
    \end{itemize}
    \end{column}
\end{columns}

\begin{frame}[fragile]\frametitle{Responsible Use: Avoid These Misuses}

\begin{itemize}
\item \textbf{Factual Accuracy:} Always verify AI-generated facts: especially numbers, dates, and citations. AI can be confidently wrong (``hallucination'').
\item \textbf{Sensitive Topics:} Avoid prompting AI for medical diagnoses, legal advice, or financial decisions: AI is not a licensed professional.
\item \textbf{Misinformation:} Do not use AI to generate misleading content, fake quotes, or deceptive text.
\item \textbf{Children's Safety:} Do not build AI applications targeting minors without appropriate safeguards.
\item \textbf{Privacy:} Do not include personally identifiable information (names, emails, phone numbers) in prompts sent to third-party AI services.
\item \textbf{Intellectual Property:} AI-generated content may inadvertently reproduce copyrighted material: review before publishing.
\item \textbf{Over-reliance:} Use AI as a first draft or idea generator, not as the final authority.
\end{itemize}	 

\end{frame}

\begin{frame}[fragile]\frametitle{Limitations: Hallucination in Action}

\textbf{Hallucination:} The AI generates plausible-sounding but completely made-up information.

\medskip
\emph{Boie is a real company. ``AeroGlide UltraSlim Smart Toothbrush'' does not exist. Ask the AI about it anyway:}

\textbf{Key lesson:} The AI does not know what it does not know. It will fill gaps with plausible-sounding fiction. Always fact-check critical information.

\textbf{Mitigation:} Use RAG to ground the AI in verified documents, or add ``If you are not certain, say so.'' to your prompt.

\begin{lstlisting}[language=Python]
prompt = f"""
Tell me about AeroGlide UltraSlim Smart Toothbrush by Boie.
"""
response = get_completion(prompt)
print(response)
# Output: A detailed, confident, completely fabricated 
#         product description: features, pricing, etc.
\end{lstlisting}

\begin{frame}[fragile]\frametitle{Take Aways}

\textbf{Prompt Engineering is an Iterative Process:}

\begin{itemize}
\item \textbf{Try something}: write your first prompt and run it.
\item \textbf{Analyse failures}: where does the output differ from what you expected? Be specific.
\item \textbf{Refine}: clarify the instruction, add an example, specify the format, add constraints.
\item \textbf{Test on known cases}: build a small set of test inputs with expected outputs and verify every new prompt version against them.
\item \textbf{Repeat}: even expert prompt engineers iterate 5--10 times before settling on a production prompt.
\end{itemize}

\medskip
\textbf{The best prompt you can write today is not the best prompt you will write next month.}

\end{frame}
