\begin{frame}[fragile]\frametitle{Saaras v3: Basic Transcription}
\textbf{Supported formats:} \texttt{wav, mp3, aac, ogg, flac, mp4/m4a, amr}\\
\textbf{Max REST duration:} 30 seconds per request.
\begin{lstlisting}
from sarvamai import SarvamAI

client = SarvamAI(api_subscription_key="YOUR_KEY")

# Mode 1: transcribe (default) - returns text in spoken language
with open("hindi_audio.wav", "rb") as audio_file:
    response = client.speech_to_text.transcribe(
        file=audio_file,
        model="saaras:v3",
        mode="transcribe",
        language_code="hi-IN"   # or omit for auto-detect
    )
print("Transcript:", response.transcript)

# Mode 2: translate - transcribe AND translate to English
with open("hindi_audio.wav", "rb") as audio_file:
    response = client.speech_to_text.transcribe(
        file=audio_file,
        model="saaras:v3",
        mode="translate"    # returns English text
    )
print("English:", response.transcript)
\end{lstlisting}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Saaras v3: All 5 Output Modes}
\textbf{When to use each mode:}
\begin{center}
\begin{tabular}{ll}
\texttt{transcribe} & Call centre transcripts \\
\texttt{translate} & Multilingual dashboards needing English output \\
\texttt{verbatim} & Legal / medical records (no normalisation) \\
\texttt{translit} & SMS, Roman-script input systems \\
\texttt{codemix} & Social media, youth audience apps \\
\end{tabular}
\end{center}
\begin{lstlisting}
modes = ["transcribe", "translate", "verbatim",
         "translit", "codemix"]

for mode in modes:
    with open("hindi_audio.wav", "rb") as audio_file:
        response = client.speech_to_text.transcribe(
            file=audio_file,
            model="saaras:v3",
            mode=mode
        )
    print(f"[{mode}] {response.transcript}")

# Expected output for "Main office ja raha hoon":
# [transcribe] मैं ऑफिस जा रहा हूँ
# [translate]  I am going to the office
# [verbatim]   मैं ऑफिस जा रहा हूँ (word-for-word)
# [translit]   Main office ja raha hoon
# [codemix]    Main office जा रहा हूँ
\end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Real-Time STT via WebSocket}
For live voice applications (voice bots, live captioning), use the WebSocket API:

\textbf{WebSocket constraint:} Only \texttt{wav} and raw PCM formats (\texttt{pcm\_s16le, pcm\_l16, pcm\_raw}) supported for streaming. Sample rate: 16kHz.
\begin{lstlisting}
import asyncio
from sarvamai import AsyncSarvamAI

async def stream_transcription():
    client = AsyncSarvamAI(api_subscription_key="YOUR_KEY")

    async with client.speech_to_text.stream(
        model="saaras:v3",
        mode="transcribe",
        language_code="hi-IN"
    ) as stream:
        # Push audio chunks (16kHz, PCM 16-bit WAV)
        with open("live_audio.wav", "rb") as f:
            while chunk := f.read(4096):
                await stream.send_audio(chunk)

        await stream.end_stream()

        # Receive transcripts as they arrive
        async for event in stream:
            print(event.transcript)

asyncio.run(stream_transcription())
\end{lstlisting}
\end{frame}

from sarvamai import SarvamAI
import os

client = SarvamAI(api_subscription_key=os.environ["SARVAM_API_KEY"])

# Step 1: Transcribe user's speech (any Indian language)
with open("user_query.wav", "rb") as f:
    stt = client.speech_to_text.transcribe(
        file=f, model="saaras:v3", mode="transcribe"
    )
user_text = stt.transcript
detected_lang = stt.language_code   # auto-detected language
print(f"[STT] ({detected_lang}): {user_text}")

# Step 2: Get LLM response (sarvam-m is FREE)
llm_resp = client.chat.completions(
    messages=[
        {"role": "system", "content":
         f"You are a helpful assistant. Reply in {detected_lang}."},
        {"role": "user", "content": user_text}
    ],
    model="sarvam-m"  # free model
)
reply_text = llm_resp.choices[0].message.content
print(f"[LLM]: {reply_text}")

# Step 3: Speak the response in the user's language
tts = client.text_to_speech.convert(
    target_language_code=detected_lang,
    text=reply_text, model="bulbul:v3", speaker="priya"
)
with open("agent_response.wav", "wb") as f:
    f.write(bytes(tts.audios[0]))
print("Response audio saved!")
\end{lstlisting}

import base64, httpx, os

SARVAM_KEY = os.environ["SARVAM_API_KEY"]

# Read and encode the PDF
with open("hindi_document.pdf", "rb") as f:
    pdf_b64 = base64.b64encode(f.read()).decode()

response = httpx.post(
    "https://api.sarvam.ai/v1/parse",
    headers={"api-subscription-key": SARVAM_KEY},
    json={"file": pdf_b64, "file_type": "pdf"}
)

data = response.json()
print("Extracted text:\n", data["text"])
# Structured page-by-page output:
for i, page in enumerate(data.get("pages", []), 1):
    print(f"\n--- Page {i} ---")
    print(page["text"])

    file_paths=["call1.mp3", "call2.mp3"],

job.wait_until_complete()

    print("Job failed!")
else:

    job.download_outputs(output_dir=str(output_dir))

import json
from pathlib import Path
from sarvamai import SarvamAI

client = SarvamAI(api_subscription_key="YOUR_KEY")

# Parse the diarized JSON output
for json_file in Path("transcriptions").glob("*.json"):
    with open(json_file) as f:
        data = json.load(f)

    entries = data.get("diarized_transcript", {}).get("entries", [])
    conversation = "\n".join(
        f"{e['speaker_id']}: {e['transcript']}"
        for e in entries

    print(conversation[:300])

        messages=[

             "You are a call analytics expert."},
            {"role": "user", "content":
             f"Analyse this call:\n{conversation}\n\n"
             "Identify: 1) Which speaker is agent vs customer? "
             "2) Was the issue resolved? 3) Customer sentiment?"}
        ],
        model="sarvam-m"  # free!
    )
    print("\nAnalysis:", resp.choices[0].message.content)

from dotenv import load_dotenv
from livekit.agents import JobContext, WorkerOptions, cli
from livekit.agents.voice import Agent, AgentSession
from livekit.plugins import openai, sarvam

load_dotenv()

class TutorAgent(Agent):
    def __init__(self):
        super().__init__(
            instructions="""You are a friendly tutor.
            Explain concepts clearly in the student's language.
            Support Hindi, Gujarati, Telugu, Bengali, Marathi.""",
            stt=sarvam.STT(
                language="unknown",    # auto-detect
                model="saaras:v3",
                mode="transcribe"
            ),
            llm=openai.LLM(model="gpt-4o"),
            tts=sarvam.TTS(
                target_language_code="hi-IN",
                model="bulbul:v3",
                speaker="anand"
            ),
        )
    async def on_enter(self):
        self.session.generate_reply()

async def entrypoint(ctx: JobContext):
    session = AgentSession()
    await session.start(agent=TutorAgent(), room=ctx.room)

if __name__ == "__main__":
    cli.run_app(WorkerOptions(entrypoint_fnc=entrypoint))
\end{lstlisting}

\begin{frame}[fragile]\frametitle{Key Limits and Tips}
\textbf{Rate Limits (Starter / free plan):}
\begin{center}
\begin{tabular}{|l|l|}
\hline
\textbf{API} & \textbf{Provisioned (Starter)} \\
\hline
STT REST & 60 req/min \\
STT WebSocket & 20 concurrent \\
Batch STT & 20 req/min \\
TTS REST & 60 req/min (Bulbul v3: 30) \\
Translation & 60 req/min \\
Chat (sarvam-m) & 60 req/min \\
Chat (30B/105B) & 40 req/min \\
Document OCR & 10 req/min (all plans) \\
\hline
\end{tabular}
\end{center}

\textbf{Key tips:}
  \begin{itemize}
    \item STT REST max: \textbf{30 seconds} per request. Use Batch API for longer audio.
    \item Transliteration API max: \textbf{1000 characters} per request. Split long texts.
    \item Transliteration between two Indic scripts (e.g., Hindi $\to$ Bengali) is \textbf{not} supported.
    \item WebSocket streaming STT only accepts \textbf{WAV/PCM} at \textbf{16kHz}.
    \item For Batch API status polling, add a \textbf{5ms minimum delay} between polls.
    \item Use \texttt{language\_code="unknown"} to auto-detect language in STT.
    \item Document OCR: max \textbf{10 pages per request}.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Exercises for Self-Study}
  \begin{itemize}
    \item \textbf{Exercise 1:} Build a Python script that accepts a WAV file from the command line, auto-detects the language, transcribes it, and saves both the transcript and a translation to English.
    \item \textbf{Exercise 2:} Create a multilingual FAQ bot: user asks a question in any Indian language (via text input), translate to English, answer with Sarvam-M LLM (free), translate answer back, and speak it with Bulbul v3.
    \item \textbf{Exercise 3:} Process a Kannada PDF (e.g., a government circular). Extract text, translate each paragraph to English, and save both versions side by side.
    \item \textbf{Exercise 4:} Build a simple call quality tool: upload 3 sample MP3 recordings, run Batch STT with diarization, and print a summary table (speaker talk-time, sentiment, issue type) using the free Sarvam-M LLM.
    \item \textbf{Exercise 5:} Extend the LiveKit voice agent to a \textit{Government Scheme Agent} that explains one state govt. scheme in the language the user speaks.
    \item \textbf{Exercise 6:} Experiment with \texttt{numerals\_format} and \texttt{spoken\_form} in transliteration. Generate TTS for ``Your account balance is Rs. 5,32,450'' in three different languages and compare.
  \end{itemize}
\end{frame}
