\begin{frame}[fragile]\frametitle{Speech to Text: Saaras v3}
\textbf{Five output modes} --- one model, many use cases:
  \begin{itemize}
    \item \textbf{\texttt{transcribe}}: Return text in the original spoken language (ASR).
    \item \textbf{\texttt{translate}}: Transcribe and translate to English in one step.
    \item \textbf{\texttt{verbatim}}: Word-for-word transcript with no post-processing.
    \item \textbf{\texttt{translit}}: Romanize the transcript (e.g., Hindi $\to$ Roman script).
    \item \textbf{\texttt{codemix}}: Handle code-mixed speech (``Hinglish'', ``Tanglish'').
  \end{itemize}

Supported audio formats: \texttt{wav, mp3, aac, aiff, ogg/opus, flac, mp4/m4a, amr}.\\
Max duration per REST request: \textbf{30 seconds}. Use Batch API for longer files.

\begin{lstlisting}
from sarvamai import SarvamAI
client = SarvamAI(api_subscription_key="YOUR_KEY")
# Transcribe a Hindi audio file
response = client.speech_to_text.transcribe(
    file=open("hindi_audio.wav", "rb"),
    model="saaras:v3",
    mode="transcribe",   # or translate/verbatim/translit/codemix
    language_code="hi-IN")  # or omit for auto-detection
print(response.transcript)
\end{lstlisting}

\end{frame}

from sarvamai import SarvamAI
client = SarvamAI(api_subscription_key="YOUR_KEY")
response = client.text_to_speech.convert(
    target_language_code="hi-IN",
    text="नमस्ते! मैं आपकी कैसे मदद कर सकता हूँ?",
    model="bulbul:v3",
    speaker="anand",   # Male voice
    # speaker="priya", # Female voice
    speech_sample_rate=24000,  # 8000/16000/24000 Hz
    enable_preprocessing=True)
# Save the audio
with open("output.wav", "wb") as f:
    f.write(bytes(response.audios[0]))
\end{lstlisting}

\begin{frame}[fragile]\frametitle{Translation and Transliteration}
\textbf{Translation with register control:}
\begin{lstlisting}
response = client.text.translate(
    input="Please submit your application form.",
    source_language_code="en-IN",
    target_language_code="gu-IN",
    mode="formal",   # formal | colloquial | modern-colloquial
    # formal: official docs/legal
    # colloquial: chatbots/social media
    # modern-colloquial: business emails
    numerals_format="native"  # native (Gujarati numerals)
                               # or international (0-9)
)
# Transliteration (script conversion, not translation):

# Hindi Devanagari -> Roman script
response = client.text.transliterate(
    input="नमस्ते, मेरा नाम विनायक है।",
    source_language_code="hi-IN",
    target_language_code="en-IN",
    spoken_form=True,    # convert to spoken form
    numerals_format="international"
)
print(response.transliterated_text)
# Output: "namaste, mera naam vinayak hai."
\end{lstlisting}

\begin{frame}[fragile]\frametitle{Voice Agent Architecture}
Building a full multilingual voice agent requires three layers:
\begin{center}
\begin{tabular}{|l|l|l|}
\hline
\textbf{Layer} & \textbf{Sarvam Model} & \textbf{Role} \\
\hline
Speech Input & Saaras v3 & Converts spoken Indian language to text \\
Understanding & Sarvam-30B / 105B & Processes query, generates response \\
Speech Output & Bulbul v3 & Converts response text to natural speech \\
\hline
\end{tabular}
\end{center}

\textbf{Integration frameworks supported:}
  \begin{itemize}
    \item \textbf{LiveKit:} Real-time voice agents. Sarvam STT and TTS plugins available via \texttt{livekit-agents[sarvam]}.
    \item \textbf{Pipecat:} Pipeline-based voice agents. \texttt{SarvamSTTService} and \texttt{SarvamTTSService} in Pipecat SDK.
    \item \textbf{Direct WebSocket:} Real-time streaming STT and TTS via Sarvam's own WebSocket API (ultra-low latency).
    \item \textbf{REST API:} Simple synchronous calls for non-real-time pipelines.
  \end{itemize}

\textbf{Pro Tip:} Use \texttt{language="unknown"} in Saaras STT to auto-detect the language. Essential for diverse Indian customer bases where users switch languages.
\end{frame}

\begin{frame}[fragile]\frametitle{Batch Processing: Call Analytics}

Pipeline: Batch STT $\to$ Parse speaker turns $\to$ LLM analysis $\to$ Sentiment/Resolution summary.


\textbf{Batch STT API:} Process up to 20 audio files (up to 1 hour each) asynchronously.
\begin{lstlisting}
from sarvamai import SarvamAI
client = SarvamAI(api_subscription_key="YOUR_KEY")
# Create a batch job with diarization
job = client.speech_to_text_translate_job.create_job(
    model="saaras:v3",
    mode="translate",        # translate to English
    with_diarization=True,   # speaker labels
)
job.upload_files(file_paths=["call1.mp3", "call2.mp3"])
job.start()
job.wait_until_complete()   # blocking; checks status
# Download results
job.download_outputs(output_dir="./transcriptions/")
# Diarization output, labels each line by speaker:
SPEAKER_00: Hello, I'm calling about my EMI payment.
SPEAKER_01: Sure, can I have your account number?
SPEAKER_00: It's 9876543210.
\end{lstlisting}
