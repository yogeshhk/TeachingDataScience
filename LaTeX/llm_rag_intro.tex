\begin{frame}[fragile]\frametitle{Motivating Questions}
  \begin{itemize}
    \item \textbf{Domain Adaptation:} How do we make a general-purpose LLM answer questions about \textit{my particular domain, say, healthcare}?
    \item \textbf{Custom LLM Solutions:} Can we build a chatbot that ``knows'' about our internal knowledge base \textit{without retraining the whole model}?
    \item \textbf{Private Data Access:} How can an LLM-based chatbot answer accurately from data it has \textit{never seen during training}?
  \end{itemize}
  
  \vspace{0.3em}
  \textbf{These are the core problems RAG is designed to solve.}
  
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Need for Domain Adaptation}

\textbf{Why can't we just use a general LLM directly?}

\begin{itemize}
\item \textbf{Privacy \& Confidentiality:} Enterprise data (contracts, emails, reports) cannot be sent to public cloud models.
\item \textbf{Knowledge Staleness:} LLMs are trained on a fixed snapshot of the web; they know nothing about yesterday's news or your internal data.
\item \textbf{Hallucinations:} LLMs confidently fabricate facts when they lack grounding in real documents.
\item \textbf{Cost of Retraining:} Training a 7B-parameter model from scratch costs \textbf{millions of dollars} and weeks of GPU time.
\end{itemize}

\vspace{0.3em}
\textbf{Key insight:} We want the \textit{reasoning ability} of a large LLM, but grounded in \textit{our own up-to-date documents}. Retraining from scratch is not the answer.

\end{frame}

\begin{frame}[fragile]\frametitle{Three Approaches to Domain Adaptation}
  \begin{itemize}
    \item \textbf{Few-Shot Prompting:} Include 2--5 worked examples directly in the prompt. 
    \newline \textit{Simple but limited by context window; does not scale to large knowledge bases.}
    
    \item \textbf{RAG (Retrieval-Augmented Generation):} At query time, \textit{fetch} the relevant documents from a database and inject them into the prompt. 
    \newline \textit{No retraining required; works with any LLM; knowledge stays current.}
    
    \item \textbf{Fine-Tuning (SFT):} Re-train the LLM on domain-specific examples to bake knowledge into its weights. 
    \newline \textit{Expensive; knowledge becomes stale; useful for style/tone adaptation.}
  \end{itemize}
  
  \vspace{0.3em}
  \textbf{RAG is the most practical and cost-effective starting point for most enterprises.}
\end{frame}

\begin{frame}[fragile]\frametitle{RAG \& SFT: Complementary Techniques}
% \begin{itemize}
  % \item RAG \& SFT are complementary, not competing methods.
  % \item RAG enhances non-parametric memory without changing parameters.
  % \item SFT changes parameters, impacting parametric memory.
% \end{itemize}
\textbf{RAG Features}
\begin{itemize}
  \item Connects to dynamic external data sources.
  \item Reduces hallucinations.
  \item Increases transparency in source of information.
  \item Works well with very large foundation models.
  \item Does not impact style, tone, vocabulary.
\end{itemize}
\textbf{SFT Features}
\begin{itemize}
  \item Changes style, vocabulary, tone of foundation model.
  \item Can reduce model size.
  \item Useful for deep domain expertise.
  \item May not address hallucinations.
  \item No improvement in transparency (black box models).
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Other Considerations}
\begin{itemize}
  \item \textbf{Latency:} RAG introduces inherent latency due to search and retrieval.
  \item \textbf{Scalability:} RAG pipelines are modular and scalable; SFT requires retraining.
  \item \textbf{Cost:} Both methods require upfront investment; costs vary.
  \item \textbf{Expertise:} RAG pipelines are moderately simple with frameworks; SFT needs deep understanding and training data creation.
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{RAG in Plain English: An Analogy}

\textbf{Think of RAG like an ``Open-Book Exam'':}

\begin{itemize}
  \item \textbf{Closed-Book LLM:} The model must answer only from what it memorized during training. If the answer is not in its memory, it may guess (hallucinate).
  
  \item \textbf{Open-Book RAG:} Before answering, the model \textit{searches} a library of documents, reads the relevant pages, and then constructs a grounded answer.
\end{itemize}

\vspace{0.5em}
\textbf{The three steps of RAG:}
\begin{enumerate}
  \item \textbf{Retrieve:} Find the most relevant document chunks from your knowledge base.
  \item \textbf{Augment:} Inject those chunks into the LLM's prompt as context.
  \item \textbf{Generate:} The LLM produces an answer grounded in the retrieved context.
\end{enumerate}

\vspace{0.3em}
{\small The LLM's \textit{reasoning} is its own; the \textit{facts} come from your documents.}

\end{frame}

\begin{frame}[fragile]\frametitle{RAG Components}


\begin{itemize}
  \item Ingestion:
	\begin{itemize}
	  \item Documents undergo segmentation into chunks, and embeddings are generated from these chunks, subsequently stored in an index.
	  \item Chunks are essential for pinpointing the relevant information in response to a given query, resembling a standard retrieval approach.
	  \end{itemize}

  \item Retrieval:
	\begin{itemize}
	  \item Leveraging the index of embeddings, the system retrieves the top-k documents when a query is received, based on the similarity of embeddings.
	\end{itemize}
	  
  \item Synthesis:
	\begin{itemize}
	  \item Examining the chunks as contextual information, the LLM utilizes this knowledge to formulate accurate responses.
	\end{itemize}

\end{itemize}

\end{frame}

\begin{frame}[fragile]\frametitle{Real-World RAG Applications}

\begin{itemize}
\item \textbf{Enterprise Q\&A:} Chatbots that answer from internal wikis, HR policies, and legal contracts (e.g., Notion AI, Confluence AI).
\item \textbf{Customer Support:} Grounded responses from product manuals and FAQs, reducing hallucination risk.
\item \textbf{Medical/Legal Research:} Retrieve clauses from case law or clinical trial papers before generating summaries.
\item \textbf{Code Assistance:} Retrieve relevant API docs and code snippets, then generate accurate code (e.g., GitHub Copilot).
\item \textbf{Personalized Education:} Fetch curriculum-specific notes before answering student questions.
\end{itemize}

\vspace{0.3em}
\textbf{Common theme:} Whenever the answer requires \textit{specific, up-to-date, or private} information, RAG is the right tool.

\end{frame}

\end{itemize}
