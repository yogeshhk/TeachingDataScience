\begin{frame}[fragile]\frametitle{Why Now? The Strategic Imperative}
  \begin{itemize}
    \item AI literacy is no longer optional, it is foundational
    \item Industry expects AI-literate graduates across all branches
    \item Peer institutions are already implementing AI programs
    \item Student employability directly linked to AI exposure
    \item Delayed action creates widening competitive gap
    \item NAAC accreditation may increasingly evaluate AI readiness
    \item NEP 2020 emphasizes technology integration across disciplines	
  \end{itemize}
\end{frame}
 
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Vision and Strategic Approach}
  \begin{itemize}
    \item AI for all disciplines, not only CSE/AI branches
    \item Teacher-first multiplier model: empower faculty to scale impact
    \item Every teacher uses AI productively and responsibly
    \item Every department integrates AI where relevant
    \item Students graduate with practical AI exposure
    \item Institute becomes recognized AI-enabled ecosystem/campus
    % \item Self-sustaining culture of innovation and adoption
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Three-Tier AI Development Model}
      \begin{itemize}
        \item Progressive pathway: Literacy → Proficiency → Expertise
        \item Level 1: AI Literacy Drive (AILD) - All faculty
        \item Level 2: AI Proficiency Drive (AIPD) - Builders
        \item Level 3: AI Expertise Drive (AIED) - Researchers
        \item Each level has distinct targets and outcomes
        % \item Structured capability building across institution
      \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Level 1: AI Literacy Drive (AILD)}
  \begin{itemize}
    \item Target: All faculty across all departments
    \item Delivery: Online self-paced certification (1-2 hours) or Youtube + quiz
    \item Key capabilities: Prompt engineering, AI-assisted teaching workflows
    \item Applications: Lecture prep, assessments, research summaries, reports
    \item Critical principle: Never use confidential data in public LLMs
    \item Success metrics: 80\% faculty certified, one workflow demo per faculty
    \item Timeline: Complete baseline certification in 3 months
    \item Foundation for institution-wide AI adoption
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Level 2: AI Proficiency Drive (AIPD)}
  \begin{itemize}
    \item Target: Technical faculty and student innovation teams
    \item Objective: Build institutional AI applications
    \item Key technologies: RAG systems, chatbots, workflow automation
    \item Sample projects: Admission assistant, library search, placement tools
    \item Infrastructure: Shared GPU server, AI sandbox, cloud credits
    \item Expected outcome: 5-10 deployed AI applications
    \item Move from usage to creation and problem-solving
    \item Participation in national AI hackathons and competitions
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Level 3: AI Expertise Drive (AIED)}

    \item Activities: Research groups, industry collaborations, funded projects
    \item Expected outcomes: Scopus/WoS publications, patents, grants

  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Implementation Timeline}
  \begin{itemize}
    \item Phase 1 (Months 0-3): Foundation
    \item Form steering committee, identify champions, launch AILD
    \item Target: 80\% faculty baseline certification
    \item Phase 2 (Months 3-6): Building
    \item Launch AIPD workshops, establish AI lab
    \item Start faculty-student projects, conduct internal hackathon
    \item Phase 3 (Months 6-12): Excellence
    \item Launch AIED research clusters, industry partnerships
    \item Deploy institutional AI tools, publish research papers
  \end{itemize}
\end{frame}
