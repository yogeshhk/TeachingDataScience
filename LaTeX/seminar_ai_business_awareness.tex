\begin{frame}[fragile]\frametitle{Session 4.1: Identifying AI Opportunities}

    \begin{itemize}
      \item \textbf{Goal:} Spot functional challenges that AI can address.
      \item Apply the \textbf{4-Lens Framework:}
        \begin{itemize}
          \item Efficiency – time/cost savings.
          \item Experience – customer and employee delight.
          \item Decision-making – improved insights.
          \item Innovation – new products/services.
        \end{itemize}
      \item Group brainstorming by function (sales, ops, HR, etc.).
    \end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Session 4.2: Evaluating Feasibility and Impact}

    \begin{itemize}
      \item Classify opportunities by timeline:
        \begin{itemize}
          \item Short-term (3–6 months)
          \item Medium-term (6–12 months)
          \item Long-term (future bets)
        \end{itemize}
      \item Introduce:
        \begin{itemize}
          \item \textbf{ROI Lens:} Efficiency, customer impact, revenue.
          \item \textbf{Change Lens:} People, process, tech shifts needed.
        \end{itemize}
      \item Use impact-effort matrices to prioritize.
    \end{itemize}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Session 4.3: Presenting AI Opportunity Canvases}

    \begin{itemize}
      \item Each group presents top 1–2 AI opportunity ideas.
      \item Discuss functional readiness and expected impact.
      \item Capture key success metrics:
        \begin{itemize}
          \item Efficiency gain (\%)
          \item Experience score improvement
          \item Potential revenue contribution
        \end{itemize}
      \item Facilitator synthesizes top enterprise themes.
    \end{itemize}

\end{frame}
