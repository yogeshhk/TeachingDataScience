\begin{frame}[fragile]\frametitle{The Agent's Fundamental Game Loop}
      \begin{itemize}
        \item Not a one-and-done action, but a continuous reasoning loop
        \item Similar to a programming \texttt{while} loop that keeps iterating
        \item \textbf{Thought}: Analyzes the situation and plans the next step
        \item \textbf{Action}: Calls a specific tool to execute the plan
        \item \textbf{Observation}: Examines results of the action taken
        \item Cycle repeats: Thought $\rightarrow$ Action $\rightarrow$ Observation
        \item Continues until the task is completely accomplished
        \item This loop enables continuous adaptation and problem-solving
      \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Two Ways Agents Use Tools}
      \begin{itemize}
        \item \textbf{JSON Agent}: Outputs a structured work-order (JSON) for another system to execute
        \begin{itemize}
            \item Requires an external system to parse and run the JSON
            \item More portable, but adds a layer of indirection
        \end{itemize}
        \item \textbf{Code Agent}: Directly writes and runs Python (or other) code
        \begin{itemize}
            \item More direct and powerful, code can do loops, conditionals
            \item Taps into thousands of existing libraries and APIs directly
            \item Easier to debug; results are grounded in actual execution
        \end{itemize}
        \item Both patterns are valid; choice depends on safety and capability needs
      \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Why Evals Are Hard to Design Upfront}
      \begin{itemize}
        \item Agentic workflows have many \textbf{unforeseen} failure modes
        \item Difficult to anticipate issues before real deployment
        \item Example: a customer order inquiry agent might unexpectedly mention competitors
        \item Problems often surface only after observing real outputs
        \item Best practice: \textbf{build first, evaluate later}
        \item Iterative inspection drives meaningful improvement
      \end{itemize}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Subjective Evals with LLMs as Judges}
      \begin{itemize}
        \item Many evaluation criteria are inherently \textbf{subjective}
        \item Free-text outputs resist simple rule-based scoring
        \item Use an \textbf{LLM as a judge} for quality assessment
        \item Example: prompt a judge LLM to rate essay quality from 1 to 5
        \item Apply the judge across many generated outputs to track trends
        \item Scores can indicate whether iterations are improving the system
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Limitations of LLM-Based Scoring}
      \begin{itemize}
        \item LLMs are \textbf{weak at fine-grained numeric ratings}
        \item 1--5 scale scores are often inconsistent and unreliable
        \item Technique is useful as an \textbf{initial rough baseline} only
        \item Better evaluation methods exist (e.g., pairwise comparison)
        \item Still useful for quick, directional comparisons early in development
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Types of Agentic AI Evaluations}
      \begin{itemize}
        \item \textbf{Objective evals}: code-based checks (did the agent mention a competitor?)
        \item \textbf{Subjective evals}: LLM judges for quality, tone, helpfulness
        \item \textbf{End-to-end evals}: measure full agent output quality for a complete task
        \item \textbf{Component-level evals}: assess individual steps (retrieval, tool call, generation)
        \item Using different eval types helps \textbf{isolate and fix} specific weaknesses
      \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{The ReAct Paper: Why It Matters}
\begin{itemize}
    \item \textbf{ReAct} (Yao et al., 2022) formalised the \textbf{Reason + Act} loop
    \item Key insight: interleaving \emph{thought traces} with \emph{tool actions} outperforms either alone
    \item Standard prompting: only generates text
    \item Chain-of-Thought: reasons but does not act
    \item ReAct: reasons \textbf{and} acts, then observes results and reasons again
    \item Results showed $\sim$34\% improvement on HotpotQA vs.\ CoT alone
    \item Intuition: thinking aloud helps the agent \textbf{stay on track} across many steps
\end{itemize}
\end{frame}

% SLIDES 57-59 — Mitigating challenges — kept, minor polish on capitalisation
\begin{frame}[fragile]\frametitle{Mitigating Predictability Challenges}
    \begin{itemize}
        \item Choose \textbf{reasoning models} when possible (e.g., Gemini 2.5 Flash, o1)
        \item Ground the LLM in \textbf{trusted data} to reduce hallucinations (RAG, search)
        \item Guide the agent to \textbf{think step-by-step} (chain-of-thought, few-shot prompting)
        \item Use \textbf{state management} and memory to track progress
        \item Implement checks in the workflow: tool-call validation, safety filters, logic guards
        \item Continuously \textbf{test and evaluate} your agent on representative scenarios
    \end{itemize}
	
		{\tiny (Ref: Google Cloud Labs H2 India Agents for All)}

\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Mitigating Stability Challenges}
    \begin{itemize}
        \item Use consistent tool abstractions via \textbf{Model Context Protocol (MCP)}
        \item Write \textbf{detailed system prompts}, include precise tool descriptions
        \item Implement \textbf{error-handling}: circuit-breakers, retries, human hand-off
        \item Utilise \textbf{state management} (especially workflow status) to recover from interruptions and prevent infinite loops
        \item Be willing to refactor as framework features and tool interfaces evolve
    \end{itemize}
	
		{\tiny (Ref: Google Cloud Labs H2 India Agents for All)}

\end{frame}

\begin{frame}[fragile]\frametitle{Mitigating Operations Challenges}
    \begin{itemize}
        \item In the dev phase: \textbf{log verbosely}, you will need the traces
        \item Build robust \textbf{test and evaluation datasets}; automate tests on every commit
        \item Gather key metrics: token throughput, response code distribution; create alerts
        \item Leverage framework \textbf{tracing features} to find and address latency bottlenecks
    \end{itemize}
