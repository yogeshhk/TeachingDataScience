\begin{frame}[fragile]\frametitle{Data Science: New?}
	\begin{itemize}
	\item Data analysis has been around for a while; statistics for centuries. Old wine in a new bottle?
	\item Fisher (1935, ANOVA), Deming (1939, quality control), Tukey (1977, exploratory data analysis) all came well before the term ``Data Science'' did.
	\item What's new is the scale: the internet (Google, 1996) and the resulting data deluge (2010 onward).
	\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Types of Data}
	\begin{itemize}
	\item \textbf{Categorical (Qualitative)}: Nominal and Ordinal
		\begin{itemize}
		\item \textbf{Nominal} (no sense of order/rank): eye color, zip codes
		\item \textbf{Ordinal} (sense of order/rank): rankings, size in \{small, medium, large\}
		\end{itemize}
	\item \textbf{Numeric (Quantitative)}: Interval and Ratio
		\begin{itemize}
		\item \textbf{Interval}: equal differences are meaningful, but no true zero. eg calendar dates, temperature
		\item \textbf{Ratio}: has a true zero. eg counts, height, weight, time
		\end{itemize}
	\end{itemize}
NOIR: \textbf{N}ominal, \textbf{O}rdinal, \textbf{I}nterval, \textbf{R}atio
\end{frame}

\begin{frame}[fragile]\frametitle{Data Engineering: Activities and Roles}
	\begin{itemize}
	\item Architecting distributed systems, creating reliable pipelines, combining data sources, architecting data stores
	\item Collaborating with data science teams and building the right solutions for them
	\item \textbf{Roles}: Generalist (small teams); Pipeline-centric (medium teams, transforms data for analysis); Database-centric (large teams, ETL into warehouses)
	\end{itemize}
