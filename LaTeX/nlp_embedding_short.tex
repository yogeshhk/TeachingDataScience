%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Why do we use word embeddings?}
	\begin{itemize}
	\item Computers cannot process words -- they need numbers.
	\item Especially for Machine Learning, where we apply mathematical rules and matrix operations to inputs.
	\item Word Embeddings are texts converted into numbers -- most ML and almost all DL architectures cannot process raw strings.
	\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{What is a Word Embedding?}
	\begin{itemize}
	\item Word embedding = vector representation of a word, `embedded' in $n$-dimensional space.
	\item Once words are vectors, their properties can be utilized -- e.g. dot product giving cosine similarity.
	\item Goal: capture some sort of relationship in that space -- meaning, morphology, context.
	\item $n$ can be tens or hundreds, instead of the millions needed for one-hot encoding.
	\item Based on Zellig Harris' ``distributional hypothesis'': words used close to one another typically have similar meaning.
	\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Overview of Different Types of Word Vectors}
	\begin{itemize}
	\item \textbf{(Traditional) Frequency-based}: One-hot (Bag of Words), TF-IDF, Co-occurrence
	\item \textbf{(Modern) Prediction-based}: Word2Vec (Google), GloVe (Stanford), BERT, \ldots
	\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{One-hot Encoding}
	\begin{itemize}
	\item A vector with as many dimensions as the corpus has unique words.
	\item Each unique word gets a unique dimension, represented by a 1 in that dimension with 0s everywhere else.
	\end{itemize}
\begin{center}
\includegraphics[width=0.5\linewidth,keepaspectratio]{w2v1}
\end{center}
Result: huge, sparse vectors that capture no relational information.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{TF-IDF Transform}
	\begin{itemize}
	\item Words are represented by term frequency $\times$ inverse document frequency.
	\item Common words (``the'', ``and'') get down-weighted; words distinctive to a subset of documents get up-weighted.
	\end{itemize}
\begin{center}
\includegraphics[width=0.45\linewidth,keepaspectratio]{tfidf1}
\end{center}
Still suffers from high dimensionality and doesn't capture semantic relatedness.
\end{frame}

\begin{frame}[fragile]\frametitle{Were These Good Vector Representations?}
	\begin{itemize}
	\item To have a ``semantic'' (meaning-wise) representation, similar words should be close together in the hyper-dimensional space.
	\item Non-similar words should be far apart.
	\item Traditional one-hot vectors (Apple = [1,0,0], Orange = [0,1,0], Plane = [0,0,1]) fail this test -- very few cells participate in the representation, and there's no notion of ``closeness''.
	\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Benefits of Word2Vec}
	\begin{itemize}
	\item Low space and time complexity to generate a rich representation.
	\item Larger dimensionality gives more features in the representation.
	\item Can efficiently train on billion-word corpora, while keeping dimensionality small.
	\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{GloVe}
	\begin{itemize}
	\item Uses global context across the entire corpus, unlike Word2Vec's local window.
	\item Builds a word-context co-occurrence matrix, then uses neural methods to decompose it into dense word vectors.
	\end{itemize}
\begin{center}
\includegraphics[width=0.6\linewidth,keepaspectratio]{w2v5}
\end{center}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{FastText -- Subword Embeddings}
	\begin{itemize}
	\item Facebook's extension to Word2Vec (2017).
	\item Breaks words into character n-grams (3--6 chars) and sums them into a whole-word feature.
	\item Handles out-of-vocabulary words -- e.g. ``circumnavigate'' borrows meaning from ``navigate''.
	\item Pre-trained vectors available in 294 languages.
	\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Domain Understanding}
	\begin{itemize}
	\item A big advantage: pre-trained embeddings are applicable across many tasks.
	\item Caution: generic-corpus embeddings carry generic meanings.
	\item ``Shell'' in zoology means a sea creature's house; in CAD it means a hollowing operation. ``Draft'' in legal writing is a rough document; in CAD it's a tapering operation.
	\item For domain-specific meanings, you need domain-specific embeddings.
	\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Key Takeaways}
\begin{enumerate}
\item Embeddings convert text to numerical representations.
\item Context is crucial for capturing meaning.
\item Evolution: Frequency-based $\rightarrow$ Prediction-based $\rightarrow$ Contextual.
\item Modern embeddings capture rich semantic relationships, with wide applications across NLP.
\item Consider domain specificity, bias/fairness, and computational resources.
\item Future: multimodal, efficient, specialized embeddings.
\end{enumerate}
\end{frame}
