\begin{frame}[fragile]\frametitle{Functions}

\item Lots of research needed to come up with equations.

    \begin{column}[T]{0.5\linewidth}
		\begin{itemize}
		\item \textbf{Artificial Intelligence (AI):} Machines mimicking human intelligence.
		\item \textbf{Machine Learning (ML):} Learning from data without explicit programming.
		\item \textbf{Deep Learning (DL):} Neural networks with many layers.
		\item \textbf{Generative AI (GenAI):} DL models that can \textit{create} new content (text, code, images) rather than just analyzing existing data.

\item 1997: Deep Blue beats Kasparov

\item 2017: Transformers introduced (Attention is All You Need)
\item 2018: BERT, GPT
\item 2020: GPT-3
\item 2022: ChatGPT launches
\item 2023: GPT-4, Claude 3, Gemini
\item 2024: Multimodal models mainstream
\item 2025: AI assistants everywhere
\end{itemize}
