%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{What Standard Neural Networks Cannot Do}
\begin{columns}
\begin{column}{0.5\textwidth}
\textbf{Standard networks work great for:}
\begin{itemize}
  \item Images (regular pixel grid)
  \item Text (ordered sequence)
  \item Tabular data (fixed-size rows)
\end{itemize}
\vspace{0.3cm}
These all live in \textbf{flat, Euclidean space} --- rows, columns, pixels in a grid.
\end{column}
\begin{column}{0.5\textwidth}
\textbf{But what about\ldots}
\begin{itemize}
  \item Will this molecule cure cancer?
  \item Is this bank transaction fraudulent?
  \item What is the 3D shape of this protein?
  \item Which road is fastest right now?
  \item Who will buy this product next?
\end{itemize}
\end{column}
\end{columns}
\vspace{0.3cm}
\begin{alertblock}{The Core Problem}
These datasets are \textbf{not grids}. They are \textbf{graphs, surfaces, and manifolds} --- and standard networks throw away their structure entirely.
\end{alertblock}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Real-World Wins with Geometric Deep Learning}
\begin{itemize}
  \item \textbf{AlphaFold (DeepMind, 2021):} Predicted 3D shapes of 200 million proteins using geometry-aware networks. Solved a 50-year biology challenge --- structure determines drug targets.
  \vspace{0.1cm}
  \item \textbf{Drug Discovery:} GNNs screen millions of molecular graphs to shortlist drug candidates in hours, not years.
  \vspace{0.1cm}
  \item \textbf{Recommendation Systems:} Pinterest, Netflix, and Uber Eats model user--item interactions as graphs for much better suggestions.
  \vspace{0.1cm}
  \item \textbf{Fraud Detection:} Banks model transaction networks as graphs; GNNs catch fraud patterns invisible in flat tables.
  \vspace{0.1cm}
  \item \textbf{Traffic Prediction:} Google Maps routes traffic over a graph (road intersections = nodes, roads = edges) for real-time ETA.
\end{itemize}
\vspace{0.2cm}
\begin{block}{}
\centering {\em All of these treat data as graphs or curved surfaces. That is Geometric Deep Learning.}
\end{block}
\end{frame}
