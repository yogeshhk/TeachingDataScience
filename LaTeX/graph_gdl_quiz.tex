\begin{frame}[fragile]\frametitle{Quiz 1: Geometric Priors}
  \begin{enumerate}
    \item Which geometric prior does a CNN primarily encode?
      \begin{itemize}
        \item[a)] Permutation invariance
        \item[b)] \textbf{Translational equivariance} \textit{(correct)}
        \item[c)] Rotational invariance
        \item[d)] Scale separation
      \end{itemize}
    \vspace{0.4cm}
    \item What three geometric priors does GDL encode as inductive biases?
      \begin{itemize}
        \item[a)] Translation, rotation, scaling
        \item[b)] \textbf{Symmetry \& invariance, stability, multi-scale representations} \textit{(correct)}
        \item[c)] Grids, graphs, gauges
        \item[d)] Manifolds, point clouds, topological spaces
      \end{itemize}
  \end{enumerate}
\end{frame}

\begin{frame}[fragile]\frametitle{Quiz 2: Symmetry and Frameworks}

      \begin{itemize}
        \item[a)] Riemannian metrics
        \item[b)] Fourier transforms
        \item[c)] \textbf{Symmetry groups and invariants under transformations} \textit{(correct)}
        \item[d)] Gradient descent
      \end{itemize}
    \vspace{0.4cm}
    \item In GNNs, which property ensures the output is unchanged when node indices are shuffled?
      \begin{itemize}
        \item[a)] Translational equivariance
        \item[b)] Stability
        \item[c)] \textbf{Permutation invariance} \textit{(correct)}
        \item[d)] Gauge equivariance
      \end{itemize}
  \end{enumerate}
\end{frame}

\begin{frame}[fragile]\frametitle{Quiz 3: GNNs and Message Passing}

    \item Which GNN type uses attention weights to decide how much to trust each neighbour?

        \item[d)] Spectral GNN
