\begin{frame}[fragile]\frametitle{Basic Agent Example - Simple Setup}
      \begin{itemize}
	\item Create a basic agent with Gemini 2.0 Flash model
	\item Define tools using Python functions with type hints
	\item Agent automatically handles function calling and response generation
	\item Clean separation between tool definition and agent logic
	\item Synchronous execution for simple use cases
	  \end{itemize}
      
      \begin{lstlisting}[language=python, basicstyle=\tiny]
# agent.py
from google.adk.agents import Agent
from google.adk.runners import Runner
from google.adk.sessions import InMemorySessionService
from google.genai import types
import yfinance as yf, asyncio

def get_stock_price(symbol: str) -> dict:
    """Get current stock price for a given symbol."""
    stock = yf.Ticker(symbol)
    return {"symbol": symbol, "price": stock.info.get('currentPrice')}

root_agent = Agent(
    name="finance_agent",
    model="gemini-2.0-flash",
    instruction="You are a helpful financial assistant. Use tools when needed.",
    tools=[get_stock_price],
)

session_svc = InMemorySessionService()
runner = Runner(agent=root_agent, app_name="demo", session_service=session_svc)

async def ask(query: str):
    s = await session_svc.create_session(app_name="demo", user_id="u1")
    msg = types.Content(role="user", parts=[types.Part(text=query)])
    async for ev in runner.run_async(user_id="u1", session_id=s.id, new_message=msg):
        if ev.is_final_response():
            print(ev.content.parts[0].text)

asyncio.run(ask("What is NVIDIA's current stock price?"))
      \end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Structured Output with Pydantic}
      \begin{itemize}
	\item Define expected output schema to ensure type-safe and structured responses
	\item Automatic validation ideal for integration with downstream systems
	  \end{itemize}
      
      \begin{lstlisting}[language=python, basicstyle=\tiny]
from google.adk.agents import Agent
from google.adk.runners import Runner
from google.adk.sessions import InMemorySessionService
from pydantic import BaseModel, Field
from google.genai import types
import asyncio, yfinance as yf

def get_stock_price(symbol: str) -> float:
    """Get current stock price."""
    return yf.Ticker(symbol).info.get('currentPrice', 0)
def get_company_info(symbol: str) -> dict:
    """Get company name, sector, and business summary."""
    t = yf.Ticker(symbol); return {"name": t.info.get('longName')}

class StockAnalysis(BaseModel):
    symbol: str = Field(description="Stock ticker symbol")
    current_price: float = Field(description="Current stock price")
    recommendation: str = Field(description="Buy/Hold/Sell recommendation")
    reasoning: str = Field(description="Analysis reasoning")

agent = Agent(
    name="analyst", model="gemini-2.0-flash",
    instruction="Analyze the stock and return structured JSON matching the schema.",
    tools=[get_stock_price, get_company_info],
    output_schema=StockAnalysis,   # enforces structured JSON output
    disallow_transfer_to_parent=True, disallow_transfer_to_peers=True)

session_svc = InMemorySessionService()
runner = Runner(agent=agent, app_name="analysis", session_service=session_svc)

async def analyze(query: str):
    s = await session_svc.create_session(app_name="analysis", user_id="u1")
    msg = types.Content(role="user", parts=[types.Part(text=query)])
    async for ev in runner.run_async(user_id="u1", session_id=s.id, new_message=msg):
        if ev.is_final_response():
            result = StockAnalysis.model_validate_json(ev.content.parts[0].text)
            print(f"Symbol: {result.symbol}  Price: ${result.current_price}")
            print(f"Rec: {result.recommendation}  - {result.reasoning}")

asyncio.run(analyze("Analyze NVIDIA stock"))
      \end{lstlisting}
\end{frame}

\begin{frame}[fragile]\frametitle{Error Handling \& Reliability}
% Fallback decorator keeps tool errors from crashing the agent; standard
% try/except around runner handles model-level failures.
      \begin{lstlisting}[language=python, basicstyle=\tiny]
from google.adk.agents import Agent
from google.adk.runners import Runner
from google.adk.sessions import InMemorySessionService
from google.genai import types
import asyncio, logging, yfinance as yf

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def safe_tool_wrapper(func):
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except Exception as e:
            logger.error(f"Tool {func.__name__} failed: {e}")
            return {"error": str(e), "fallback": True}
    return wrapper

@safe_tool_wrapper
def get_stock_price(symbol: str) -> dict:
    """Get current stock price for a ticker symbol."""
    return {"price": yf.Ticker(symbol).info['currentPrice']}

agent = Agent(name="resilient_agent", model="gemini-2.0-flash",
    instruction="Handle errors gracefully and inform users.",
    tools=[get_stock_price])

session_svc = InMemorySessionService()
runner = Runner(agent=agent, app_name="demo", session_service=session_svc)

async def ask(query: str):
    s = await session_svc.create_session(app_name="demo", user_id="u1")
    msg = types.Content(role="user", parts=[types.Part(text=query)])
    try:
        async for ev in runner.run_async(user_id="u1", session_id=s.id, new_message=msg):
            if ev.is_final_response():
                print(ev.content.parts[0].text)
    except Exception as e:
        logger.error(f"Runner error: {e}")
        print("Sorry, I'm having trouble processing your request.")

asyncio.run(ask("What's the price of NVDA?"))
      \end{lstlisting}
\end{frame}
