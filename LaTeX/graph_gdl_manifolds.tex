\begin{frame}[fragile]\frametitle{High-Dimensional Representation}
    \begin{itemize}
        \item A 2D point is represented by two values (x, y). They can also be represented by two boxes --- the first box's darkness depends on the x coordinate, the second on y. 0 = white, 1 = black.
        \item A 16D point uses 16 coordinate values, visualised as a 4$\times$4 grid of boxes (a tiny image).
    \end{itemize}

\begin{frame}[fragile]\frametitle{Images as Points in a Hypercube}
    \begin{itemize}
        \item A grayscale image = a single point in a very high-dimensional space.
        \item A 256$\times$256 image has 65,536 dimensions --- one per pixel.
        \item All images are just rare points in this vast space.
        \item A random point in this space is almost certainly noise, not a meaningful image.
		\item So only a tiny subset of all possible points in this high-dimensional space are ``valid'' images.
    \end{itemize}

\begin{frame}[fragile]\frametitle{Transition Between Images}
    \begin{itemize}
        \item \textbf{Straight-line interpolation} between two face images produces weird, blurry in-between images that look like nothing real.
        \item \textbf{Smooth curved paths exist} where every intermediate point is also a valid, realistic face.
        \item These paths lie on the \textbf{image manifold} --- e.g. morphing smoothly from a smiling face to a neutral face.
    \end{itemize}

\begin{frame}[fragile]\frametitle{Understanding the Face Manifold}
    \begin{itemize}
        \item Each realistic face image lies \emph{on or very near} a manifold in pixel space.
        \item Moving \textbf{along} the manifold changes expressions, lighting, or pose --- smoothly and realistically.
        \item Moving \textbf{off} the manifold leads to unrealistic or noisy images.
        \item The face manifold is \textbf{low-dimensional} (a few degrees of freedom) embedded in a very high-dimensional pixel space.
    \end{itemize}

\begin{frame}[fragile]\frametitle{Manifolds in Ambient Space}
    \begin{itemize}
        \item Data may lie on a \textbf{1D manifold} (a curve) embedded in 2D space.
        \item The manifold is the ``true structure''; the extra dimensions are noise.
        \item Manifolds are locally homeomorphic to Euclidean space --- every small patch looks flat.
        \item Local coordinates on the manifold are all we need to describe the data.
    \end{itemize}

\begin{frame}[fragile]\frametitle{Degrees of Freedom and Natural Data}
    \begin{itemize}
        \item Real-world data has constraints: limited \textbf{degrees of freedom}.
        \item Example: human height and weight are correlated --- not truly independent dimensions.
        \item Physical laws restrict images: no cubical planets, no levitating objects.
        \item Natural images cluster on specific low-dimensional manifolds within pixel space.
    \end{itemize}

\begin{frame}[fragile]\frametitle{Exploring Manifolds Visually}
    \begin{itemize}
        \item Moving \textbf{on} a manifold preserves realistic image features.
        \item Deviating \textbf{from} the manifold introduces noise or anomalies.
        \item Clock images example: midpoints on-manifold give a plausible intermediate clock; midpoints off-manifold give garbage.
        \item On-manifold interpolation = meaningful generation.
    \end{itemize}

\begin{frame}[fragile]\frametitle{Learning the Manifold}
    \begin{itemize}
        \item ML models aim to \textbf{approximate} data manifolds.
        \item VAEs and GANs learn a mapping from a \emph{low-dimensional latent space} to pixel space.
        \item The latent space is low-dimensional (e.g. 128 numbers); the image manifold lives in millions of dimensions.
        \item The model learns this mapping through training --- not manual rules.
    \end{itemize}

\begin{frame}[fragile]\frametitle{Manifolds vs.\ Graphs: The Big Picture}
    \begin{itemize}
        \item Both are ways to describe \textbf{non-Euclidean structure} in data.
        \item \textbf{Manifolds} are \emph{continuous} spaces that locally look like flat Euclidean space.
        \item \textbf{Graphs} are \emph{discrete} structures made of nodes and edges.
        \item Manifolds generalise smooth surfaces and curves.
        \item Graphs represent relationships or connections --- no notion of smoothness required.
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{What is a Manifold?}
    \begin{itemize}
        \item A manifold locally looks like $\mathbb{R}^n$ (flat Euclidean space).
        \item Every small neighbourhood on the manifold is approximately flat.
        \item Tangent space and curvature can be defined on smooth manifolds.
        \item Examples: the surface of a sphere, a torus (donut surface), a curved road in 3D space.
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{What is a Graph?}
    \begin{itemize}
        \item Graphs consist of \textbf{vertices (nodes)} and \textbf{edges (links)}.
        \item No notion of continuity or smoothness --- purely combinatorial.
        \item Structure depends entirely on how nodes are connected.
        \item Used to represent social networks, road maps, molecules, knowledge bases, etc.
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Connecting Manifolds and Graphs}
    \begin{itemize}
        \item Graphs can \textbf{approximate manifolds}: sample points on a surface and connect nearby ones.
        \item Manifolds can be thought of as continuous analogs of graphs.
        \item Concepts often translate: geodesic distance on a manifold $\leftrightarrow$ shortest path on a graph.
        \item Many GDL methods exploit this duality (e.g. graph Laplacian approximates the Laplace-Beltrami operator on manifolds).
    \end{itemize}
\end{frame}
