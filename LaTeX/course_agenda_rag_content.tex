\begin{frame}[fragile]\frametitle{From Zero to Hero in RAG}
\framesubtitle{Then from RAG to Riches}

\textbf{Learning Path:} Python $\rightarrow$ DSA $\rightarrow$ System Design $\rightarrow$ NLP $\rightarrow$ ML $\rightarrow$ LLMs $\rightarrow$ RAG $\rightarrow$ Production
\vspace{0.3cm}  

\begin{columns}
    \begin{column}[T]{0.5\linewidth}
      \textbf{Full-Time Mode}
      \begin{itemize}
        \item Duration: 15 working days (3 weeks)
        \item Daily: 8 hours (4 learning + 4 coding)
        \item Week 4: Independent capstone
        \item Saturdays: Overview + QnA (2 hrs)
      \end{itemize}
    \end{column}
    \begin{column}[T]{0.5\linewidth}
      \textbf{Part-Time Mode}
      \begin{itemize}
        \item Duration: 12 weeks (3 months)
        \item Weekly: Self-paced tutorials + coding
        \item Month 4: Independent capstone
        \item Weekends: Optional QnA sessions
      \end{itemize}
    \end{column}
  \end{columns}

\vspace{0.3cm}  
\textbf{What You'll Build:}
\begin{itemize}
  \item 12-15 progressive projects (choose 1 per day/week)
  \item 1 production-ready capstone system
  \item Complete end-to-end RAG portfolio
\end{itemize}

\vspace{0.3cm}
\textbf{Target Audience:} Software engineers with basic programming knowledge
\end{frame}
