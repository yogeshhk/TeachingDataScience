\begin{frame}[fragile]\frametitle{What Is the Core Idea?}
At the heart of all AI is one desire:
\begin{center}
\textbf{Find a function $f$ that maps inputs $X$ to outputs $Y$}
\end{center}
\begin{itemize}
\item $f(x) = 2x + 3$, simple, hand-coded rule
\item $f(\text{email text}) = \text{spam or not spam}$, needs learning from data
\item $f(\text{photo}) = \text{face identity}$, too complex for rules, needs Deep Learning
\item $f(\text{``write me a summary''}) = \text{paragraph}$, needs a Large Language Model
\end{itemize}
\textbf{Intuition:} Traditional programming is \textit{you telling the machine the rules}. Machine Learning is \textit{the machine figuring out the rules from examples}.
\end{frame}

\begin{frame}[fragile]\frametitle{What is AI?, A Working Definition}
\textbf{Artificial Intelligence (AI):}
\begin{itemize}
\item Machines performing tasks that \textit{normally require human intelligence}
\item Thinking, reasoning, recognizing patterns, learning, generating content
\end{itemize}

\textbf{My definition:} ``If a computer program starts doing intelligent tasks, perceiving, reasoning, deciding, creating, that's AI.''

\textbf{Key Differentiators from traditional software:}
\begin{itemize}
\item Traditional software: \textit{explicit rules} $\rightarrow$ output
\item AI/ML: \textit{data + algorithm} $\rightarrow$ the rules emerge automatically
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Types of Machine Learning}
\begin{itemize}
\item \textbf{Supervised Learning:} Learn from \textit{labeled} examples (input + correct answer)
  \begin{itemize}
  \item Example: Email $\rightarrow$ spam / not-spam; image $\rightarrow$ tumor / benign
  \end{itemize}
\item \textbf{Unsupervised Learning:} Find hidden structure in \textit{unlabeled} data
  \begin{itemize}
  \item Example: Cluster customers into segments without pre-defined groups
  \end{itemize}
\item \textbf{Reinforcement Learning:} Agent learns by \textit{reward and penalty} from actions
  \begin{itemize}
  \item Example: AlphaGo mastered chess by playing millions of games against itself
  \end{itemize}
\item \textbf{Self-supervised / Foundation Models:} Learn from raw internet text by predicting missing words, the basis of all modern LLMs
\end{itemize}
\end{frame}

  \textbf{Reality:} LLMs predict the \textit{statistically most likely next token}, they do not "understand."

\item \textbf{Myth:} You need to train your own model.\\
  \textbf{Reality:} For 95\% of enterprise use cases, you \textit{integrate} existing LLMs via API + RAG.
\item \textbf{Myth:} Enterprise AI and consumer AI are the same.\\
  \textbf{Reality:} Enterprise AI needs security, data isolation, compliance, cost controls.
\end{itemize}

\begin{frame}[fragile]\frametitle{What Is a Large Language Model (LLM)?}
\textbf{How it works (college-level intuition):}
\begin{itemize}
\item An LLM is trained to predict: \textit{``Given these words, what word comes next?''}
\item Mathematically: given token sequence $w_1, w_2, \ldots, w_n$, compute $P(w_{n+1} \mid w_1 \ldots w_n)$
\item \textbf{What this means in plain English:} The model reads billions of web pages and books, and builds an internal ``map'' of which words and ideas go together. When you ask it a question, it generates the most likely continuation of your text.
\item During training on trillions of tokens, the model learns grammar, facts, reasoning patterns, and code, all from next-word prediction alone.
\end{itemize}
\textbf{Size matters:} ``Large'' = billions of parameters (weights). GPT-4 $\approx$ 1 trillion parameters.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{How Transformers Work, The Intuition}
\textbf{The key innovation: Attention Mechanism}
\begin{itemize}
\item Before Transformers: RNNs read text word-by-word, forgetting early context
\item Transformers read the \textit{entire sequence at once} and compute which words to \textit{pay attention to}
\item \textbf{Formula:} $\text{Attention}(Q, K, V) = \text{softmax}\!\left(\frac{QK^T}{\sqrt{d_k}}\right)V$
\item \textbf{What this means:} For each word, calculate a ``relevance score'' with every other word, then mix their meanings proportionally. The word ``bank'' in ``river bank'' pays more attention to ``river'' than ``money.''
\end{itemize}
\textbf{Result:} Transformers handle long documents, code, and multi-turn conversations much better than older models.
\end{frame}

\begin{frame}[fragile]\frametitle{Small LLMs (SLLMs), When Smaller Is Better}
\textbf{What is an SLLM?} A compact model (1B--13B parameters) designed for specific tasks or on-device deployment.

\begin{itemize}
\item \textbf{Examples:} Phi-3 Mini, Gemma 2B, Llama 3.2 3B, Mistral 7B
\item \textbf{When to use SLLM over LLM:}
  \begin{itemize}
  \item Latency is critical (real-time, embedded systems)
  \item Data must not leave your network (on-premise requirement)
  \item Task is narrow and well-defined (e.g., classify support ticket type)
  \item Cost is a primary constraint (1000$\times$ cheaper per token)
  \end{itemize}
\item \textbf{When to use a large LLM:}
  \begin{itemize}
  \item Complex multi-step reasoning required
  \item Long context window needed (legal docs, large codebases)
  \item You need the best possible quality without cost sensitivity
  \end{itemize}
\end{itemize}
\textbf{Rule:} Start with the smallest model that meets your quality bar.
\end{frame}

\begin{frame}[fragile]\frametitle{What Is an AI Agent?}
\textbf{Chatbot vs.\ Agent, the critical distinction:}
\begin{itemize}
\item \textbf{Chatbot:} One-shot Q\&A. User asks $\rightarrow$ LLM replies. No memory, no tools, no action.
\item \textbf{Agent:} LLM given \textit{tools} (web search, code execution, APIs, databases) and a \textit{goal}. It \textbf{plans}, executes steps, observes results, and adapts, autonomously.
\end{itemize}

\textbf{Agent Loop (ReAct pattern):}
\begin{center}
Observe $\rightarrow$ Think $\rightarrow$ Act $\rightarrow$ Observe $\rightarrow$ \ldots $\rightarrow$ Goal Achieved
\end{center}

\textbf{Engineering examples:}
\begin{itemize}
\item ``Analyze this failing build, find root cause in logs, create a JIRA ticket.'', 3 tool calls, no human in loop
\item ``Review this PR for security issues, check OWASP top 10, comment inline.'', Agent with code analysis tool
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{MCP, Model Context Protocol}
\textbf{What is MCP?} An open standard (Anthropic, 2024) that defines how LLMs connect to external tools and data sources.

\begin{itemize}
\item \textbf{Problem before MCP:} Every AI tool had custom integrations. Fragile, hard to maintain.
\item \textbf{MCP solution:} Standardized protocol, LLM $\leftrightarrow$ Tool (like USB for AI)
\item \textbf{MCP architecture:}
  \begin{itemize}
  \item \textbf{MCP Host:} The AI app (e.g., Claude, Cursor)
  \item \textbf{MCP Server:} A connector to a specific tool (GitHub, Jira, Postgres, filesystem)
  \item \textbf{MCP Client:} Bridge inside the host that speaks the protocol
  \end{itemize}
\item \textbf{Engineering relevance:} Build one MCP server for your internal tool, it works with any MCP-compatible AI agent
\end{itemize}

\textbf{Analogy:} Before: proprietary chargers for every phone. After: USB-C for all.
\end{frame}

\begin{frame}[fragile]\frametitle{Key Security Risks in Enterprise AI}
\begin{itemize}
\item \textbf{Data Leakage:} Developer pastes internal credentials or PII into ChatGPT, it may be used for training
  \begin{itemize}
  \item \textit{Mitigation:} Use enterprise API (data not used for training), DLP tools on clipboard
  \end{itemize}
\item \textbf{Prompt Injection:} Malicious input tricks the AI agent into performing unauthorized actions
  \begin{itemize}
  \item \textit{Mitigation:} Input sanitization, tool-call allowlists, human approval for destructive actions
  \end{itemize}
\item \textbf{Hallucinations in Production:} Model confidently returns wrong API schema, wrong config
  \begin{itemize}
  \item \textit{Mitigation:} Retrieval-Augmented Generation (RAG) grounding, human review for critical decisions
  \end{itemize}
\item \textbf{Supply Chain Risk:} Malicious or poisoned open-source model weights
  \begin{itemize}
  \item \textit{Mitigation:} Only use models from verified registries (Hugging Face verified, official providers)
  \end{itemize}
\end{itemize}

\begin{frame}[fragile]\frametitle{Why RAG? The Core Problem}
\textbf{The problem with raw LLMs:}
\begin{itemize}
\item LLMs are trained on public internet data up to a cutoff date
\item They know \textit{nothing} about your internal runbooks, your codebase, your product docs
\item Fine-tuning is expensive (\$10k--\$100k+) and needs to be re-done for every update
\end{itemize}

\textbf{RAG = Retrieval-Augmented Generation}
\begin{itemize}
\item Idea: At query time, \textit{retrieve} relevant documents from your knowledge base and \textit{inject} them into the LLM prompt as context
\item The LLM now has your private data, without fine-tuning, without data leakage
\item Result: LLM answers based on \textit{your} documents, not just its training data
\end{itemize}

\textbf{Analogy:} Imagine giving an expert consultant your internal wiki right before the meeting, instead of hoping they already know it.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{RAG Architecture, End to End}
\textbf{Indexing Phase (one-time):}
\begin{enumerate}
\item Load documents (PDFs, Confluence, code, Slack threads)
\item Split into chunks (e.g., 500 tokens each with 50-token overlap)
\item Convert each chunk to a vector embedding: $\vec{v} = \text{Embed}(\text{chunk})$\\
  \textit{What is an embedding? A list of 1500 numbers that captures meaning. Similar text $\rightarrow$ similar numbers.}
\item Store vectors + original text in a Vector Database
\end{enumerate}

\textbf{Query Phase (real-time):}
\begin{enumerate}
\item User asks: ``How do I restart the payment service?''
\item Convert query to embedding: $\vec{q} = \text{Embed}(\text{query})$
\item Retrieve top-$k$ most similar chunks (cosine similarity: $\cos(\vec{q}, \vec{v}) = \frac{\vec{q} \cdot \vec{v}}{|\vec{q}||\vec{v}|}$)\\
  \textit{This measures the ``angle'' between two meaning vectors. Closer angle = more relevant.}
\item Inject retrieved chunks + user query into LLM prompt $\rightarrow$ Get grounded answer
\end{enumerate}
\end{frame}

\begin{frame}[fragile]\frametitle{How LLMs ``Think'', Tokens and Context}
\begin{itemize}
\item LLMs process text as \textbf{tokens}, roughly 3/4 of a word on average
\item The \textbf{context window} is how much text the model can ``see'' at once, 128K to 1M tokens in modern models
\item Everything you want the model to know \textit{must be in the context}: history, instructions, documents
\item The model is \textit{stateless} by default, it forgets everything between API calls
\end{itemize}

\textbf{Practical implication:}
\begin{itemize}
\item Put the most important instructions at the \textit{beginning} and \textit{end} of the prompt (recency \& primacy effects)
\item Don't assume the model knows your codebase, provide relevant snippets
\item Token costs money: $\approx$ \$0.003 per 1K input tokens (GPT-4o pricing guide)
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Prompt Patterns, Role Prompting}
\textbf{Pattern:} Assign the model an expert role to activate relevant knowledge and style.

\begin{block}{Basic Prompt}
\texttt{Explain Kubernetes pod scheduling.}
\end{block}

\begin{block}{Role Prompt (Better)}
\texttt{You are a senior Site Reliability Engineer with 10 years of Kubernetes experience, mentoring a junior developer. Explain pod scheduling concisely, using an analogy.}
\end{block}

\textbf{Why it works:} The role activates specific patterns from training, an expert's vocabulary, precision, and communication style.\\
\textbf{Use in SDLC:} ``You are a security code reviewer following OWASP Top 10...''
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Prompt Patterns, Few-Shot Prompting}
\textbf{Pattern:} Provide 2--3 input/output examples before your actual request.

\begin{block}{Few-Shot for Commit Message Generation}
\texttt{Input: Added null check in user auth handler}\\
\texttt{Output: fix(auth): add null guard for missing user token}\\[0.4em]
\texttt{Input: Updated Dockerfile to use slim base image}\\
\texttt{Output: chore(docker): switch to python:3.11-slim base image}\\[0.4em]
\texttt{Input: Fixed off-by-one in pagination logic}\\
\texttt{Output:}
\end{block}

\textbf{Why it works:} Examples define the output format, style, and level of detail, far more precisely than any instruction.\\
\textbf{Use in SDLC:} Few-shot for test case format, release note style, JIRA ticket format.
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Prompt Patterns, Chain-of-Thought}
\textbf{Pattern:} Ask the model to \textit{reason step-by-step} before giving the final answer.

\begin{block}{Without CoT (bad for reasoning)}
\texttt{Is this SQL query safe from injection? [query]. Answer yes or no.}
\end{block}

\begin{block}{With Chain-of-Thought (better)}
\texttt{Review this SQL query for injection vulnerabilities. First, identify all user-controlled inputs. Second, check if they are parameterized. Third, check for dynamic string concatenation. Finally, give your verdict with justification.}
\end{block}

\textbf{Why it works:} Forces the model to generate intermediate reasoning, reducing errors on complex multi-step tasks. Think of it as asking someone to ``show their work.''
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Prompt Patterns, Structured Output}
\textbf{Pattern:} Ask for output in a specific structured format (JSON, YAML, Markdown table).

\begin{block}{Structured Output Prompt}
\texttt{Analyze this stack trace and return ONLY valid JSON:}\\
\texttt{\{"root\_cause": "...", "affected\_service": "...",}\\
\texttt{"suggested\_fix": "...", "severity": "P1|P2|P3"\}}
\end{block}

\textbf{Why it works for engineering:}
\begin{itemize}
\item Downstream code can parse and act on structured output programmatically
\item Reduces hallucinated free-text, enforces schema
\end{itemize}

\textbf{Advanced:} Use \textit{function calling / tool use} API feature to guarantee valid JSON schema from the model.
\end{frame}

\begin{frame}[fragile]\frametitle{Prompting for Engineering Tasks, Examples}

\item \textbf{Debugging:}
  \texttt{``Here is a Python function and the failing test output. Identify the bug and provide a fix with explanation.''[code][output]}
\item \textbf{Refactoring:}
  \texttt{``Refactor the following function to reduce cyclomatic complexity, improve readability, and add docstrings. Preserve behavior exactly.''[code]}

\end{itemize}

\begin{frame}[fragile]\frametitle{How to Test AI Deliverables}
Testing AI is fundamentally different from testing deterministic software.

\textbf{Key principles:}
\begin{itemize}
\item \textbf{Non-determinism:} Same input $\rightarrow$ different output each run. Use \textit{range of valid outputs}, not exact match.
\item \textbf{Evaluation (Evals):} Create a test set of (input, expected behavior) pairs. Measure:\\
  Relevance, Faithfulness (grounded in source), Toxicity, Format correctness
\item \textbf{Human-in-the-loop review:} For high-stakes outputs (code going to prod, customer communications), require human approval
\item \textbf{Regression suite:} As you improve prompts, run regression to ensure no quality degradation
\end{itemize}

\textbf{Tools:} Ragas (RAG evaluation), LangSmith (tracing + eval), PromptFoo (prompt regression testing)
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{MLOps / AI Ops, Operating AI in Production}
\textbf{What is MLOps?} The discipline of deploying, monitoring, and maintaining ML/AI systems reliably in production.

\begin{itemize}
\item \textbf{Model Versioning:} Track which model version is in production; enable rollback
\item \textbf{Prompt Versioning:} Treat prompts as code, version-controlled, reviewed, tested
\item \textbf{Observability for LLMs:}
  \begin{itemize}
  \item Latency per call, token usage, cost per user
  \item Quality metrics: hallucination rate, user satisfaction, rejection rate
  \end{itemize}
\item \textbf{Drift Detection:} Model outputs degrading over time? Monitor quality metrics, re-evaluate regularly.
\item \textbf{Guardrails:} Input/output validation to block prompt injection, off-topic responses, toxic content
\item \textbf{Canary Deployments:} Roll out new model/prompt to 5\% traffic, validate quality before 100\% rollout
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Monitoring AI Agents in Production}
Agents introduce unique operational challenges:
\begin{itemize}
\item \textbf{Trace every step:} Log each reasoning step, tool call, and result, not just the final output
\item \textbf{Tool call auditing:} Which external systems did the agent call? With what parameters? What was returned?
\item \textbf{Failure modes to monitor:}
  \begin{itemize}
  \item Infinite loops (agent stuck retrying)
  \item Hallucinated tool calls (inventing APIs that don't exist)
  \item Cascading failures (agent errors that trigger more agents)
  \end{itemize}
\item \textbf{Human escalation triggers:} Define conditions under which the agent must pause and request human approval (e.g., destructive database operations, customer-facing communications)
\item \textbf{Cost monitoring:} Multi-step agents can consume 10$\times$--100$\times$ more tokens than single-turn calls
\end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Sample Code, RAG Bot in 20 Lines}
\begin{verbatim}
from langchain.document_loaders import DirectoryLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import Chroma
from langchain.chains import RetrievalQA
from langchain.llms import OpenAI

# 1. Load your internal docs
docs = DirectoryLoader("./runbooks/").load()

# 2. Chunk and embed
chunks = RecursiveCharacterTextSplitter(
    chunk_size=500).split_documents(docs)
db = Chroma.from_documents(chunks, OpenAIEmbeddings())

# 3. Create RAG chain
qa = RetrievalQA.from_chain_type(
    llm=OpenAI(), retriever=db.as_retriever())

# 4. Answer questions
print(qa.run("How do I restart the auth service?"))
\end{verbatim}
\end{frame}

\begin{frame}[fragile]\frametitle{AI ROI, How to Think About It}
\textbf{ROI formula (simplified):}
\[
\text{ROI} = \frac{\text{Value Gained} - \text{Total AI Cost}}{\text{Total AI Cost}} \times 100\%
\]
\textit{This means: for every rupee spent on AI, how many rupees of value do you get back?}

\textbf{Value can be:}
\begin{itemize}
\item \textbf{Time saved:} 40\% reduction in PR review time for 20 engineers = $20 \times 0.4 \times \text{hourly rate}$ per week
\item \textbf{Quality gains:} Bugs caught by AI before production = avoided incidents (quantify per P1 cost)
\item \textbf{Revenue impact:} Faster feature delivery = earlier market capture
\item \textbf{Cost avoidance:} AI handles Tier-1 support = fewer support hires
\end{itemize}

\textbf{Productivity metric frameworks:} SPACE (Satisfaction, Performance, Activity, Communication, Efficiency), DORA metrics with AI
\end{frame}

\begin{frame}[fragile]\frametitle{FinOps for AI, Cost Control Strategies}
\textbf{Token cost is your primary lever, optimize it:}
\begin{itemize}
\item \textbf{Use the right model for the task:} Route simple queries to GPT-4o-mini (\$0.15/1M) not GPT-4o (\$2.50/1M), 16$\times$ cheaper
\item \textbf{Prompt caching:} Cache repeated system prompts; Anthropic offers 90\% discount on cached tokens
\item \textbf{Limit output tokens:} Set \texttt{max\_tokens}, output is 3--4$\times$ more expensive than input
\item \textbf{Batching:} Use Batch API for non-real-time tasks, 50\% cost reduction
\item \textbf{Monitoring dashboards:} Track cost per user, per feature, per team, set budgets and alerts
\item \textbf{Usage governance:} Define who can use which model tier; require justification for expensive models
\end{itemize}

\textbf{Target:} Set cost-per-interaction KPI before launching any AI feature to production.
\end{frame}

\begin{frame}[fragile]\frametitle{Key AI Risks for Engineering Leaders}
\begin{itemize}
\item \textbf{Hallucinations:} Model confidently states wrong information.
  \textit{Mitigation: RAG grounding, human review for high-stakes outputs, show sources}
\item \textbf{Data Leakage:} Sensitive data sent to external LLM APIs.
  \textit{Mitigation: DLP, approved-only APIs, no PII in prompts policy}
\item \textbf{Bias \& Unfairness:} AI trained on biased data produces biased outputs.
  \textit{Examples: Amazon hiring tool (2018), facial recognition errors for minorities}
\item \textbf{Auditability:} Cannot explain why AI made a decision, regulatory risk.
  \textit{Mitigation: Log all inputs/outputs, use explainable models where regulation demands}
\item \textbf{Regulatory Exposure:}
  \begin{itemize}
  \item EU AI Act (2024): Risk-based regulation; high-risk AI systems require conformity assessment
  \item GDPR: AI decisions using personal data require legal basis + right to explanation
  \end{itemize}
\end{itemize}
\end{frame}

\begin{itemize}

\begin{frame}[fragile]\frametitle{When to Use What, Decision Tree}
\begin{itemize}
\item \textbf{Clear rules, deterministic logic?} $\rightarrow$ Traditional programming
\item \textbf{Pattern recognition in structured data?} $\rightarrow$ Classical ML (XGBoost, sklearn)
\item \textbf{Images, speech, complex unstructured data?} $\rightarrow$ Deep Learning / CNN / RNN
\item \textbf{Text generation, Q\&A, summarization, code?} $\rightarrow$ LLM via API
\item \textbf{Q\&A on your private documents?} $\rightarrow$ RAG + LLM
\item \textbf{Multi-step autonomous task on internal systems?} $\rightarrow$ LLM Agent + MCP

\item \textbf{Real-time physical-world control?} $\rightarrow$ Physical AI / Robotics
\end{itemize}
