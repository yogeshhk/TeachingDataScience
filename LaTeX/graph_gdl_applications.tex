\begin{frame}[fragile]\frametitle{AlphaFold: GDL's Breakthrough Moment}

\begin{columns}
\begin{column}{0.6\textwidth}
\begin{itemize}
  \item \textbf{Problem:} A protein's 3D shape determines its biological function --- and shape is hard to measure experimentally (costs years, millions of dollars per protein).
  \item \textbf{AlphaFold2 (DeepMind, 2021):} Predicts 3D protein structure from amino acid sequence with near-experimental accuracy.
  \item \textbf{Key GDL ideas used:} SE(3)-equivariant attention (respects 3D rotations/translations), graph-based residue interaction modelling, iterative refinement on the structure manifold.
  \item \textbf{Impact:} 200 million protein structures predicted and released publicly. Accelerating drug discovery, vaccine design, and fundamental biology.
\end{itemize}
\end{column}
\begin{column}{0.4\textwidth}
\begin{block}{}
\centering
{\em ``A solution to a 50-year-old grand challenge in biology.''}\\
\vspace{0.2cm}
{\small --- Science Magazine, Breakthrough of the Year 2021}
\end{block}
\vspace{0.3cm}
{\small Key lesson: treating the protein as a \textbf{graph + 3D manifold} --- not a flat sequence --- was the decisive advantage.}
\end{column}
\end{columns}

\end{frame}
