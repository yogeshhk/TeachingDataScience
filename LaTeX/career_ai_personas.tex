\begin{frame}[fragile]\frametitle{The USER Persona}

% \begin{columns}
    % \begin{column}[T]{0.5\linewidth}
\textbf{Who You Are:}
Domain expert who wants to use AI, not build it from scratch.

\textbf{Modern Tools (2024):}
\begin{itemize}
\item \textbf{No-code AI Builders:} Zapier AI Actions, Make.com, n8n
\item \textbf{AI Analytics:} Julius AI, Rows, Formula Bot, DataChat
\item \textbf{Decision Intelligence:} Dataiku, Alteryx AI
\item \textbf{Document AI:} ChatPDF, Claude + PDF upload, NotebookLM
\item \textbf{Workflow Automation:} ChatGPT + Custom GPTs, Claude Projects
\end{itemize}

    % \end{column}
    % \begin{column}[T]{0.5\linewidth}

\textbf{What You Can Build:}
\begin{itemize}
\item Automated email responses (Gmail + ChatGPT)
\item Data analysis workflows (Julius AI)
\item Customer support bots (Zapier + Claude API)
\item Content generation pipelines (Make.com + GPT-4)
\end{itemize}
    % \end{column}
% \end{columns}

\vspace{0.2cm}
\textbf{Reality:} You don't need to code to leverage AI. Start here if you're non-technical or want fastest ROI.

\begin{frame}[fragile]\frametitle{The DEVELOPER Persona }

% \begin{columns}
    % \begin{column}[T]{0.5\linewidth}
\textbf{Who You Are:}
Technical professional who wants to build AI applications with code.

\textbf{Modern Stack :}
\begin{itemize}
\item \textbf{LLM APIs:} OpenAI, Anthropic, Google Gemini
\item \textbf{Orchestration:} LangChain, LlamaIndex, Haystack, Semantic Kernel
\item \textbf{Vector DBs:} Pinecone, Weaviate, Qdrant, Chroma
\item \textbf{Frameworks:} Streamlit, Gradio, Chainlit (UI)
\item \textbf{Monitoring:} LangSmith, Helicone, Arize
\item \textbf{Deployment:} Vercel AI SDK, Modal, Replicate
\end{itemize}


    % \end{column}
    % \begin{column}[T]{0.5\linewidth}
\textbf{What You'll Build:}
\begin{itemize}
\item RAG applications (Chat with your data)
\item AI agents (autonomous task execution)
\item Custom chatbots with tool calling
\item Multi-modal applications
\end{itemize}
    % \end{column}

    % \end{columns}
\end{frame}
