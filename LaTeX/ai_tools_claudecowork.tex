\begin{frame}[fragile]\frametitle{What is Claude Cowork?}
  \begin{itemize}
    \item \textbf{One-Line Definition:} Claude Code's agentic power, delivered inside Claude Desktop — no terminal required.
    \item \textbf{Origin Story:} Built after Anthropic observed non-developers using Claude Code for vacation research, building slide decks, cleaning email, recovering photos, and more — tasks far beyond coding.
    \item \textbf{What Changed:} Instead of responding one prompt at a time, Claude now \textit{takes on tasks} — plans, executes, and delivers finished work autonomously.
    \item \textbf{Built With Claude Code:} Ironically, the Cowork feature itself was built in $\approx$10 days, largely \textit{by} Claude Code.
    \item \textbf{Status:} Research Preview (Jan 2026) — macOS \& Windows (x64) · Paid plans only.
    \item \textbf{Market Signal:} Cowork's launch triggered a \$285B selloff in enterprise SaaS stocks — investors pricing in structural disruption to project management, writing, and workflow automation tools.
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Cowork vs. Chat vs. Claude Code}
  \begin{itemize}
    \item \textbf{Claude Chat (claude.ai):} One prompt → one response. Stateless. No file system access. 
    \item \textbf{Claude Code (Terminal):} Full agentic power — but requires terminal, CLI fluency, and manual environment setup. Targets developers.
    \item \textbf{Claude Cowork (Desktop):} Same agentic architecture as Claude Code — file access, multi-step execution, sub-agents — wrapped in a GUI. Targets \textit{everyone}.
    \item \textbf{The Abstraction:} Cowork is Claude Code with the terminal hidden, a filesystem sandbox pre-configured, and a task-oriented UI surfaced.
    \item \textbf{Underlying Model:} Powered by Claude Opus 4.6 — Anthropic's most capable model, with a 50\%-task-completion time horizon of 14.5 hours (METR benchmark, Feb 2026).
    \item \textbf{Tab Navigation:} In Claude Desktop: \texttt{Chat} tab → \texttt{Cowork} tab → \texttt{Tasks} mode.
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{How Cowork Executes Tasks}
  \begin{itemize}
    \item \textbf{Step 1 — Analyze:} Claude reads your task description and forms an execution plan.
    \item \textbf{Step 2 — Decompose:} Complex tasks are broken into subtasks; sub-agents are spun up as needed.
    \item \textbf{Step 3 — Execute in VM:} Code and file operations run inside an isolated virtual machine (Apple Virtualization Framework / \texttt{VZVirtualMachine}), booting a custom Linux root filesystem — your system is sandboxed.
    \item \textbf{Step 4 — Parallel Workstreams:} Sub-agents coordinate and work simultaneously on independent subtasks.
    \item \textbf{Step 5 — Deliver:} Outputs (documents, spreadsheets, organized files) land directly in your designated folder.
    \item \textbf{Transparency Throughout:} Progress indicators + live reasoning surface at every step. You can steer mid-task without restarting.
    \item \textbf{Deletion Protection:} Permanent file deletion requires explicit \texttt{Allow} confirmation — no silent data loss.
  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Folder Access \& Sandbox Model}
  \begin{itemize}
    \item \textbf{Designated Folder:} You grant Cowork access to a specific local folder — not your entire file system.
    \item \textbf{VM Isolation:} Files are mounted into a containerized VM environment. Claude cannot reach anything outside the sandbox boundary.
    \item \textbf{Read + Write:} Within the sandbox, Claude can read existing files, modify them, and create new ones — no manual upload/download.
    \item \textbf{Local Storage:} Conversation history is stored locally on your machine — not subject to Anthropic's cloud data retention policy.
    \item \textbf{Best Practice:} Create a dedicated working folder with only the files relevant to the task. Keep sensitive data (credentials, financial records) outside the granted folder.
    \item \textbf{No Audit Logs:} Cowork activity is \textit{not} captured in Anthropic's Audit Logs, Compliance API, or Data Exports — not suitable for regulated workloads.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Key Capabilities}
  \begin{itemize}
    \item \textbf{Direct Local File Access:} Read, write, create, and organize files without manual upload/download cycles.
    \item \textbf{Sub-Agent Coordination:} Claude orchestrates parallel sub-agents for complex, multi-workstream tasks.
    \item \textbf{Professional Outputs:} Generates polished deliverables — Excel spreadsheets with live formulas, PowerPoint decks, formatted Word documents, PDFs.
    \item \textbf{Long-Running Tasks:} No conversation timeouts or context-limit interruptions; tasks run for hours if needed.
    \item \textbf{Global \& Folder Instructions:} Persistent preferences (tone, output format, role context) apply to every session automatically.
    \item \textbf{Plugin Commands:} Type \texttt{/} to invoke installed plugin capabilities inline during any task.
    \item \textbf{Mid-Task Steering:} Interrupt, redirect, or add context at any point without restarting from scratch.
    \item \textbf{Internet Access:} Cowork can fetch live data, search the web, and pull from external services during task execution.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Persistent Instructions — Context That Sticks}
  \begin{itemize}
    \item \textbf{Global Instructions:} Set once in \texttt{Settings > Cowork > Global Instructions}. Apply to every Cowork session, every task.
    \item \textbf{Use Cases:}
      \begin{itemize}
        \item Preferred output format: ``Always produce executive summaries under 200 words.''
        \item Role context: ``I am a CFO at a Series B startup — prioritize cost sensitivity.''
        \item Tone: ``Write in active voice; avoid jargon.''
      \end{itemize}
    \item \textbf{Folder Instructions:} Project-specific context attached to a folder. Claude can also \textit{update} these autonomously during a session as it learns more about your project.
    \item \textbf{Why It Matters:} Eliminates repetitive context-setting. Cowork effectively ``learns'' your work style across sessions without requiring re-prompting.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Installing \& Customizing Plugins}
  \begin{itemize}
    \item \textbf{Install Flow:}
      \begin{enumerate}
        \item Open Claude Desktop → \texttt{Cowork} tab
        \item Click \texttt{Plugins} in the left sidebar
        \item Browse library → click to install, or \texttt{Upload plugin} for custom files
        \item Plugins save locally to your machine
      \end{enumerate}
    \item \textbf{Customization:} Click \texttt{Customize} on any installed plugin → Claude prompts you through tailoring it to your workflow.
    \item \textbf{Custom Plugins:} Build from scratch using the \textit{Plugin Create} plugin — no coding expertise required.
    \item \textbf{Invoking Plugins:} Type \texttt{/} or click \texttt{+} in any Cowork session to see and invoke plugin commands.
    \item \textbf{Organization Deployment:} Organization-wide plugin provisioning is on the roadmap (not yet available in research preview).
  \end{itemize}

\begin{frame}[fragile]\frametitle{Safety \& Permissions Model}
  \begin{itemize}
    \item \textbf{Filesystem Sandbox:} All file operations confined to the VM-mounted folder you designate — true OS-level isolation via Apple Virtualization Framework (macOS).
    \item \textbf{Deletion Protection:} Permanent deletion requires explicit user confirmation at runtime. No silent file loss.
    \item \textbf{Prompt Injection Risk:} Cowork has internet access and reads files — malicious content in those files could attempt to hijack task instructions. Anthropic filters known attack patterns but cannot guarantee full coverage.
    \item \textbf{Mitigation Best Practices:}
      \begin{itemize}
        \item Grant access to a dedicated, purpose-built working folder only.
        \item Keep credentials, financial records, and personal data outside the granted folder.
        \item Make task instructions as explicit and unambiguous as possible.
        \item Save backups before running destructive tasks.
        \item Monitor progress indicators — steer if the plan looks wrong.
      \end{itemize}
    \item \textbf{Not for Regulated Workloads:} Cowork activity is not logged in Audit Logs, Compliance API, or Data Exports. Unsuitable for HIPAA, FINRA, or similar regulated environments.
    \item \textbf{Network Egress:} Cowork respects your organization's existing network egress permissions.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Competitive Landscape}
  \begin{itemize}
    \item \textbf{vs. Microsoft Copilot:} Copilot is deeply native to Microsoft 365 (Word, Excel, Teams, Outlook) — unbeatable inside that ecosystem. Cowork operates at the OS level with folder access — more flexible but requires more setup. Notably, Microsoft itself has encouraged thousands of employees to adopt Claude Code and Cowork.
    \item \textbf{vs. Google Gemini:} Gemini integrates tightly into Google Workspace (Docs, Sheets, Gmail). Cowork is workspace-agnostic and local-first.
    \item \textbf{vs. OpenAI Operator/ChatGPT Agent:} OpenAI's browser agent is narrower in scope; Cowork's architecture (built on Claude Code) provides more robust, proven agentic behavior from a larger existing user base.
    \item \textbf{Cowork's Structural Edge:} Bottom-up evolution — Claude Code was hardened by millions of developer users before Cowork launched. That reliability and model quality carries over.
    \item \textbf{Market Signal:} \$285B SaaS selloff at launch reflects investor conviction that OS-level AI agents threaten entire categories of enterprise software — PM tools, writing assistants, data platforms, and workflow automation.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Getting Started — Step by Step}
  \begin{itemize}
    \item \textbf{1. Download Claude Desktop:}
\begin{lstlisting}
https://claude.com/download
(macOS universal or Windows x64)
\end{lstlisting}
    \item \textbf{2. Ensure Paid Plan:} Pro, Max, Team, or Enterprise subscription required.
    \item \textbf{3. Open Cowork Tab:} In Claude Desktop, click the \texttt{Cowork} tab.
    \item \textbf{4. Grant Folder Access:} Select a dedicated working folder — not your home directory.
    \item \textbf{5. Set Global Instructions:} \texttt{Settings > Cowork > Global Instructions} — define your role, tone, preferred output formats.
    \item \textbf{6. Install Plugins:} Browse the plugin library; start with your function (Sales, Finance, Marketing, etc.).
    \item \textbf{7. Describe a Task:} Write your goal in natural language. Be specific about the outcome, not the steps. Let Cowork plan.
    \item \textbf{8. Monitor \& Steer:} Watch progress indicators. Intervene if the plan drifts. Approve deletion operations explicitly.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Mid-Task Steering}
  \begin{itemize}
    \item \textbf{Goal:} Show that Cowork is not a black box — you stay in control.
    \item \textbf{Task:} Start a long-running document reorganization task across 50+ files.
    \item \textbf{Observation:} During execution, Cowork's plan proposes moving archive files to a subfolder you'd prefer to keep at root level.
    \item \textbf{Intervention:}
\begin{lstlisting}
> [Mid-task input]
  Stop, keep the /archive folder at root level,
  don't nest it. Continue with everything else
  as planned.
\end{lstlisting}
    \item \textbf{What Happens:} Cowork acknowledges the correction, updates its plan, and resumes without restarting.
    \item \textbf{Key Message:} Agentic $\neq$ uncontrollable. Transparency + steering gives you the benefits of automation without surrendering oversight.
  \end{itemize}
