\begin{frame}[fragile]\frametitle{What's Still Hard: Observability}
      \begin{itemize}
        \item Observability tracks agent behaviour at \textbf{each step}
        \item Essential: logs of tool calls, retries, and decisions
        \item Use metrics to catch latency and cost issues early
        \item Debug via \textbf{step-wise traceability} through reasoning chains
        \item Detect when agents ``go off-rail'' and take unexpected actions
        \item Tools like \textbf{Langfuse, Arize Phoenix, Comet Opik} support observability
        \item Must be \textbf{designed in from day one}, retrofitting is painful
        \item Critical for high-autonomy, production-deployed agent systems
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{What's Still Hard: Evaluation}
      \begin{itemize}
        \item Agents are \textbf{non-deterministic}, need \emph{ongoing} evaluation, not one-time testing
        \item Track \textbf{goal/task success rates} regularly across diverse inputs
        \item Monitor tool call success and failure rates per tool
        \item Measure RAG quality (retrieval precision, hallucination rate)
        \item Check for \textbf{overthinking or inefficiency} (excessive tool calls, loops)
        \item Analyse latency and token usage per reasoning step
        \item Avoid ``vibe checks''; use \textbf{real, quantitative metrics}
        \item Think of evaluation as \textbf{Quality Assurance for agentic systems}
      \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Agentic AI: Protocols $>$ Prompts}
\begin{columns}
    \begin{column}[T]{0.6\linewidth}
      \begin{itemize}
        \item Handcrafted prompts don't scale with system complexity
        \item Shift toward \textbf{shared protocols and open standards}
        \item \textbf{MCP} (Model Context Protocol): standardises how agents receive context (tools, memory, RAG, prior instructions)
        \item \textbf{A2A} by Google: enables agent-to-agent messaging across frameworks
        \item Shared schemas will enable cross-agent collaboration
        \item Protocols bring \textbf{cleaner, reusable abstractions}
        \item Adoption will take time, just as HTTP did for the web
      \end{itemize}
    \end{column}

\begin{frame}[fragile]\frametitle{Agentic AI: Hybrid Reasoning Models}
      \begin{itemize}
        \item Future agents will combine \textbf{deep planning} and \textbf{fast reactive actions}
        \item One model can't do everything efficiently
        \item Use reasoning when needed; act quickly otherwise
        \item Models like \textbf{Claude 3.7 Sonnet} and \textbf{Gemini 2.5 Flash} are early examples
        \item Hybrid approaches \textbf{balance intelligence and speed}
        \item Reduces unnecessary overthinking and improves resource use
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Agentic AI: Better Memory Systems}
      \begin{itemize}
        \item Memory is currently \textbf{poorly integrated} in most agent frameworks
        \item Future: smarter, context-aware memory systems
        \item Recall \emph{what matters}, \emph{when} and \emph{why} it matters
        \item Memory scoped to tasks, sessions, or personas
        \item Easier to manage, debug, and audit
        \item Critical for \textbf{long-term agent autonomy} and personalisation
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Agentic AI: Tool Ecosystem Maturity}
      \begin{itemize}
        \item Current tools are often \textbf{custom, ad hoc, and fragile}
        \item Maturing ecosystem will offer \textbf{plug-and-play APIs}
        \item Expect reusable, trusted, standardised tool interfaces (MCP is the start)
        \item Better abstraction layers and shared \textbf{security practices} will emerge
        \item Inspired by the evolution of microservices in software engineering
        \item Enables scalable and secure agent systems at enterprise scale
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[fragile]\frametitle{Safety \& Human Oversight in Agentic Systems}
      \begin{itemize}
        \item Higher autonomy = higher risk of \textbf{unintended consequences}
        \item Key safety principles for production agents:
        \begin{itemize}
            \item \textbf{Minimal footprint:} Request only necessary permissions; prefer reversible actions
            \item \textbf{Confirmation gates:} Pause and ask a human before high-impact, irreversible actions (e.g., sending emails, deleting records)
            \item \textbf{Sandboxed execution:} Run tool calls in isolated environments
            \item \textbf{Audit trails:} Log every action for post-hoc review
        \end{itemize}
        \item \textbf{Rule of thumb:} Err on the side of doing less and confirming when uncertain
        \item Safety is not optional, it is a \textbf{core design requirement}
      \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Key Principles for LLM Agent Success}
    \begin{itemize}
        \item Focus on building the \textbf{right system}, not the most sophisticated one
        \item Progress systematically:
            \begin{itemize}
                \item Start with simple prompts
                \item Optimise through comprehensive evaluation
                \item Add multi-step agents only when necessary
            \end{itemize}
        \item Core implementation principles:
            \begin{itemize}
                \item \textbf{Simplicity:} fewer moving parts, fewer failure modes
                \item \textbf{Transparency:} visible planning steps build trust
                \item \textbf{ACI:} carefully design the Agent-Computer Interface
            \end{itemize}
        \item Goal: Create powerful, \textbf{reliable}, and maintainable agents
    \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Challenges in LLM-Centred Agents}
  \begin{itemize}
    \item \textbf{Finite Context Length:} Restricted context capacity limits inclusion of history, detailed instructions, and API responses, agents can ``forget'' earlier steps
    \item \textbf{Long-term Planning:} LLMs struggle to adjust plans when faced with unexpected errors mid-task
    \item \textbf{Less Robust than Humans:} People learn efficiently from trial and error; LLMs don't update their weights at runtime
    \item \textbf{Reliability of Natural Language Interface:} LLMs may make formatting errors or refuse instructions in edge cases
  \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Decision Guide: Agents vs.\ Simpler Approaches}
\begin{columns}
    \begin{column}[T]{0.5\linewidth}
        \textbf{Use Agents when:}
        \begin{itemize}
            \item Task has \textbf{many unpredictable steps}
            \item Requires dynamic decision-making
            \item High value justifies latency/cost
            \item Errors are \textbf{easily discoverable}
            \item Examples: research, software dev, data analysis
        \end{itemize}
    \end{column}
    \begin{column}[T]{0.5\linewidth}
        \textbf{Avoid Agents when:}
        \begin{itemize}
            \item Task is simple and \textbf{well-defined}
            \item A script or workflow suffices
            \item Errors are \textbf{hard to catch} or high-stakes
            \item Deep domain expertise is required
            \item Examples: legal analysis, medical diagnosis
        \end{itemize}
    \end{column}
\end{columns}
\vspace{0.3cm}
\textbf{Golden rule:} Start simple. Add agent complexity only when simpler approaches demonstrably fail.
\end{frame}
