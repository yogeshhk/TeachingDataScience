\begin{frame}[fragile]\frametitle{LangChain: The OG Agent Framework}
      \begin{itemize}
	  \item Open-source library connecting LLMs with data and APIs
	  \item Launched late 2022; fastest-growing GitHub project by mid-2023
	  \item Provides modular components: prompts, memory, tools, interfaces
	  \item Enables everything from simple chatbots to autonomous agents
	  \item Chain-based approach for sequential operations
	  \item Foundation for many other agent frameworks (e.g., LangGraph)
	  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{LangGraph: Graph-Based Agent Orchestration}
      \begin{itemize}
	  \item Newer framework from the LangChain team, using a \textbf{graph} model
	  \item Defines \textbf{nodes} (steps) and \textbf{edges} (transitions) instead of linear chains
	  \item Enables complex workflows with \textbf{conditional branching} and loops
	  \item More low-level and controllable than high-level LangChain agents
	  \item Ideal for complex decision flows, human-in-the-loop, and retry logic
	  \item Provides granular control over agent behaviour and state
	  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{LlamaIndex: LLMs + Your Data}
      \begin{itemize}
	  \item Formerly GPT Index; specialises in \textbf{Retrieval-Augmented Generation (RAG)}
	  \item Go-to framework for connecting LLMs with external datasets
	  \item Indexes knowledge bases for intelligent, fact-grounded retrieval
	  \item Pre-built agents for question-answering scenarios
	  \item Integrates with databases, vector stores, and external APIs
	  \item Essential for document-based and enterprise knowledge systems
	  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{SmolAgents: Minimalist Code-First Approach}
      \begin{itemize}
	  \item Hugging Face framework emphasising simplicity and efficiency
	  \item Build powerful agents in just a few lines of code
	  \item \textbf{``Code-as-actions''} philosophy: agents write and execute Python directly
	  \item Up to 30\% fewer steps and API calls vs.\ verbose instruction agents
	  \item Core logic only $\sim$1{,}000 lines of code, easy to audit and extend
	  \item Lightweight and developer-friendly; quick deployment without overhead
	  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Autogen: Multi-Agent Communication}
      \begin{itemize}
	  \item Microsoft framework for \textbf{multi-agent conversations}
	  \item Models applications as conversations between specialised agents
	  \item Supports group chats, hierarchical chats, and human-in-the-loop
	  \item Agents can execute code in various sandboxed environments
	  \item Enables AI pair-programming: agents critique and refine solutions together
	  \item Ideal for tasks requiring multiple expert perspectives
	  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{LangFlow: Visual LangChain Interface}
      \begin{itemize}
	  \item Web-based graphical UI for building LangChain flows
	  \item \textbf{Drag-and-drop} components instead of writing code
	  \item Visualises agent ``thought'' processes interactively
	  \item Lowers the entry barrier for non-developers and product teams
	  \item Rapid prototyping and experimentation without boilerplate
	  \item Same underlying capabilities as LangChain, with a visual interface
	  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{CrewAI: Team-Based Agent Collaboration}
      \begin{itemize}
	  \item Framework inspired by \textbf{human team collaboration}
	  \item Sets up multiple agents with different roles in ``\textbf{crews}''
	  \item High-level configuration over manual coding
	  \item Example: Researcher agent + Writer agent working together
	  \item Abstractions for tasks, planning, and multiple memory types
	  \item Developer-friendly: common multi-agent patterns handled out-of-the-box
	  \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{n8n: Visual Workflow Automation with AI}
      \begin{itemize}
	  \item Open-source automation tool with LLM agent integration
	  \item \textbf{Drag-and-drop} workflow editor, visual Zapier with AI nodes
	  \item AI agent nodes operate within automated enterprise workflows
	  \item Maintains conversation context and calls tools across steps
	  \item Example: Auto-analyse support tickets and recommend responses
	  \item Acts as the \textbf{conductor} for LLM agents in enterprise environments
	  \end{itemize}
\end{frame}

	  \item Operates through a chat interface with high-level goals
	  \item Plans and executes tasks \textbf{autonomously} over hours or days
	  \item Internal cognition engine: API calls, web browsing, code execution
	  \item Maintains an \textbf{internal feedback loop} and memory for progress tracking
	  \item Goes beyond ChatGPT: keeps running until the task is complete

\begin{frame}[fragile]\frametitle{Build Your First AI Agent}
      \begin{itemize}
        \item Takes just 30--60 minutes to get a working agent running
        \item Follow a clear, seven-step process
        \item Focus on \textbf{functionality} over theory
        \item Each step builds on the previous one
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 1: Define a System Prompt}
      \begin{itemize}
        \item Set the agent's \textbf{goals, constraints, and behaviour}
        \item Use structured prompting: role, context, output format
        \item Be explicit about what the agent should and should \emph{not} do
        \item A clear system prompt is the single biggest lever for agent quality
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 2: Select an LLM}
      \begin{itemize}
        \item Choose a \textbf{reasoning-capable model} (e.g., Claude Sonnet 4, GPT-4o, o1-mini)
        \item Match model capability to task complexity and latency requirements
        \item Frameworks like n8n or Agno abstract the model behind a unified API
        \item Start cheap; upgrade only when quality demands it
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 3: Connect Tools}
      \begin{itemize}
        \item Add tools that match the agent's goals
        \item Common tools: web search, calculator, SQL query, file I/O, email sender
        \item Use \textbf{MCP servers} for standardised, reusable tool integration
        \item Write clear tool descriptions, the LLM uses them to decide \emph{when} to call each tool
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 4: Connect Memory}
      \begin{itemize}
        \item \textbf{Short-term memory}: in-context history of the current session
        \item \textbf{Long-term memory}: vector store, SQL, or graph database for persistence
        \item Memory lets the agent remember past actions and avoid repeated work
        \item Essential for multi-turn tasks that span more than one conversation turn
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 5: Orchestrate the Logic}
      \begin{itemize}
        \item Map out core logic that is not tied to a single agent
        \item Enable \textbf{agent-to-agent communication} for complex, multi-role tasks
        \item Choose between \textbf{static flows} (predictable) and \textbf{dynamic routing} (flexible)
        \item Move deterministic if/then rules to the orchestration layer, not the agent prompt
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 6: Add User Interface}
      \begin{itemize}
        \item Use no-code tools like \textbf{Lovable} or \textbf{Bolt} for fast UI scaffolding
        \item Alternatively, expose a \textbf{FastAPI} endpoint (e.g., via Agno's AgentOS)
        \item A good UI is critical for user trust and adoption of the agent
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Step 7: Evaluate the Agent}
      \begin{itemize}
        \item Skip fixed metrics early on, do \textbf{error analysis} instead
        \item Let performance metrics emerge from real usage data
        \item For RAG agents: evaluate \textbf{retrieval} and \textbf{generation} steps separately
        \item Build a regression test suite to prevent regressions on future updates
      \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Bottom Line: Just Start}
      \begin{itemize}
        \item Follow practical guides, many require no coding to begin
        \item Use established frameworks (n8n, Agno, LangGraph) and tools (MCP)
        \item Stop theorising about the perfect architecture, \textbf{ship} real systems
        \item Iterate fast: build $\rightarrow$ observe $\rightarrow$ evaluate $\rightarrow$ improve
      \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{RAG vs.\ Fine-Tuning: Business Impact}
    \begin{itemize}
        \item Business-critical systems require the right LLM strategy
        \item Fine-tuning feels powerful, but often adds avoidable complexity and cost
        \item Start with simpler and more flexible approaches first
        \item The decision tree: \textbf{Prompt Engineering} $\rightarrow$ \textbf{RAG} $\rightarrow$ \textbf{Fine-Tuning}
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Pre-Fine-Tuning Checklist}
    \begin{itemize}
        \item Can \textbf{prompt engineering} alone solve the task?
        \item Could \textbf{Retrieval-Augmented Generation (RAG)} help more?
        \item Is there a clear, testable baseline already in place?
        \item If all three answers are ``yes, tried it'', then consider fine-tuning
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Why RAG Outperforms Fine-Tuning in Most Cases}
    \begin{itemize}
        \item \textbf{Higher Accuracy}: Answers grounded in retrieved, relevant context
        \item \textbf{Fewer Hallucinations}: More reliable than relying on baked-in model weights
        \item \textbf{Dynamic Updates}: Update the data without retraining the model
        \item \textbf{Lower Cost}: No GPU training runs; just index and query
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{When to Use Fine-Tuning}
    \begin{itemize}
        \item RAG cannot solve context window limitations for very long documents
        \item Domain-specific \textbf{tone or writing style} is required (e.g., legal, medical)
        \item Latency is critical and retrieval adds unacceptable overhead
        \item The system is mature enough to absorb the added complexity
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Strategy Summary}
    \begin{itemize}
        \item \textbf{Prompt engineering} solves 30--50\% of tasks
        \item \textbf{RAG} adds power for another 30--40\%
        \item \textbf{Fine-tuning} is best for the final 10\% of hard, style-critical problems
        \item Always choose the \textbf{simplest effective} approach first
    \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Agentic AI Protocols Overview}
    \begin{itemize}
        \item \textbf{A2A}, \textbf{MCP}, and \textbf{ADK} are foundational for modern agentic systems
        \item Each solves a unique challenge in building autonomous agents
        \item They work \emph{together} to enable scalable multi-agent architectures
        \item Think of them as the \textbf{plumbing} beneath the agent logic
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{A2A: Agent-to-Agent Protocol}
    \begin{itemize}
        \item Enables \textbf{agent discovery, task delegation, and result exchange}
        \item Facilitates coordination in distributed, multi-agent systems
        \item Example: Agent A discovers and delegates a sub-task to Agent B; B returns results
        \item Developed by Google; enables cross-framework agent interoperability
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{MCP: Model Context Protocol}
    \begin{itemize}
        \item \textbf{Standardises} how agents access tools, APIs, and external data
        \item An open standard (Anthropic, 2024) that prevents vendor lock-in
        \item Ensures consistent, secure interaction with external systems
        \item Example: Agent queries a vector database or triggers a payment via a single MCP interface
        \item Think: \textbf{USB-C for AI agent tools}
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{ADK: Agent Development Kit}
    \begin{itemize}
        \item Toolkit for building \textbf{A2A-compliant} agents quickly
        \item Includes libraries and scaffolds for rapid agent development
        \item Compatible with frameworks like CrewAI, LangGraph, Semantic Kernel
        \item Reduces boilerplate so developers focus on agent \emph{logic}, not plumbing
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{How They Work Together}
    \begin{itemize}
        \item \textbf{MCP}: Makes each agent \emph{powerful} by giving it standardised tool access
        \item \textbf{A2A}: Makes agents \emph{collaborative} by enabling inter-agent messaging
        \item \textbf{ADK}: Makes agent development \emph{fast} with reusable scaffolding
        \item Together: a \textbf{modular, interoperable} agent ecosystem
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Real-World Analogy}
    \begin{itemize}
        \item \textbf{MCP}: The standardised tools in a mechanic's toolbox
        \item \textbf{A2A}: The communication protocol between mechanics on a team
        \item \textbf{ADK}: The training blueprint for onboarding each new mechanic fast
    \end{itemize}
\end{frame}

\begin{frame}[fragile]\frametitle{Why This Stack Matters}

        \item Quickly becoming an emerging standard for future AI workflows

\begin{frame}[fragile]\frametitle{When to Build AI Agents}
    \begin{itemize}
        \item Don't build agents for \emph{every} task, complexity has costs
        \item Ideal for complex, ambiguous, high-value problems
        \item Use deterministic \textbf{workflows} when decision paths are clear and fixed
        \item Agents consume tokens, ensure the value justifies the cost
        \item Avoid agents if error discovery is slow or the stakes are very high
        \item Good fit: \textbf{software development} (complex but verifiable)
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Designing Simple, Scalable Agents}
    \begin{itemize}
        \item Every agent = \textbf{Model + Tools + Environment}
        \item Start with the \textbf{simplest possible} components; avoid early complexity
        \item Reuse agent backbones across many use cases
        \item Don't optimise until the core behaviour is stable
        \item Visual clarity in the architecture builds trust in the system
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Optimisation \& Performance}
    \begin{itemize}
        \item \textbf{Parallelise} tool calls to reduce latency
        \item \textbf{Cache} repeated action paths in coding agents to save tokens
        \item Show \textbf{step-by-step progress} to build user trust
        \item Optimise costs only \emph{after} validating the core agent loop
        \item Keep environments simple before scaling up
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Think Like Your Agent}
    \begin{itemize}
        \item Your agent \textbf{only sees} what is in its context window
        \item Don't expect magic, expect bounded reasoning within that window
        \item Weird agent actions often mean \textbf{missing or ambiguous context}
        \item Debug by simulating the task from the agent's point of view
        \item Replay full trajectories and ask: \emph{``Why did you do that?''}
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{Tools \& Self-Improvement}
    \begin{itemize}
        \item Define tools with \textbf{clear inputs and expected outputs}
        \item Use the LLM itself to review whether tool descriptions are clear
        \item Let agents \textbf{self-critique} prompts and tool definitions
        \item Build \textbf{meta-tools} that help the agent improve its own tooling
        \item Better ergonomics reduce errors and unnecessary retries
    \end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[fragile]\frametitle{The Future: Multi-Agent \& Budget-Aware}
    \begin{itemize}
        \item Solo agents dominate now, but multi-agent is the next step
        \item \textbf{Multi-agent} = modular reasoning + parallel task execution
        \item Sub-agents help preserve the main agent's context window
        \item \textbf{Budget-awareness}: define token, time, and latency constraints upfront
        \item Role-based coordination with async interactions is the next frontier
    \end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{AutoGPT: The First Autonomous Agent}

\textbf{What it introduced (2023)}
\begin{itemize}
    \item Autonomous AI agent designed to carry out tasks \textbf{until solved}
    \item Connected to the internet for real-time research
    \item Self-prompts: breaks goals into sub-tasks and spawns child agents
    \item Over 100k GitHub stars, sparked the ``agents'' movement
\end{itemize}

\textbf{Lessons learned}
\begin{itemize}
    \item Got caught in infinite loops; sometimes hallucinated task completion
    \item Broad general scope diluted effectiveness
    \item Shifted the community toward \textbf{specialised agents} for specific tasks
    \item Opened new challenges: control, misuse prevention, unforeseen consequences
\end{itemize}

\end{frame}

\begin{frame}[fragile]
\frametitle{BabyAGI: Task-Driven Autonomous Agent}

\textbf{Architecture (4 sub-agents)}
\begin{itemize}
    \item \textbf{Execution Agent}: executes current task via GPT-4
    \item \textbf{Task Creation Agent}: generates new tasks from prior results
    \item \textbf{Prioritisation Agent}: re-ranks the task list dynamically
    \item \textbf{Context Agent}: accumulates and merges intermediate results
\end{itemize}

\textbf{Key takeaways}
\begin{itemize}
    \item Operates in an infinite loop: execute $\rightarrow$ enrich $\rightarrow$ create $\rightarrow$ prioritise
    \item Uses GPT-4 + vector database + LangChain for adaptive task management
    \item Demonstrated that \textbf{task decomposition + memory} is viable
    \item Early precursor to modern multi-agent orchestration systems
\end{itemize}

\end{frame}
